/*
 * When we have a new project we want to start with the deliverables page. This requires the project type ID, so we can fetch
 * e.g. deliverables and all other relevant info with API call.
 *
 * Once the user has clicked "Next" we need to save the project with an API call. From that point the project is referenced
 * in each ID. This ID should therefore work for direct linking and browser refresh.
 *
 * URLs starting with /new/ are for NEW project listings, and EDIT when editing a completed (draft/matching) project. In terms
 * of functionality they are identical, in that in both cases we are working with an existing project. They are however useful
 * to keep as separate sets of URLs in order to differentiate e.g. labels, headers, etc.
 *
 */
 
import store from '@/store'
 
const components = {
  deliverables: () => import(/* webpackChunkName: "listing-deliverables" */ '@/components/listing/Deliverables'),
  resources: () => import(/* webpackChunkName: "listing-resources" */ '@/components/listing/Resources'),
  impact: () => import(/* webpackChunkName: "listing-impact" */ '@/components/listing/Impact'),
  image: () => import(/* webpackChunkName: "listing-image" */ '@/components/listing/Image'),
  final: () => import(/* webpackChunkName: "listing-final-details" */ '@/components/listing/FinalDetails'),
  review: () => import(/* webpackChunkName: "listing-review" */ '@/components/listing/Review')
  // preview: () => import(/* webpackChunkName: "listing-preview" */ '@/components/listing/Preview')
}
 
// For now, if the project is not found, just redirect to dashboard
export const fetchProject = async (to, from, next) => {
  let project = store.getters['listing/project']
  if (!project || parseInt(to.params.id) !== project.id) {
    project = await store.dispatch('listing/fetchProject', to.params.id)
  }
  if (!project) {
    window.location.href = '/dashboard'
  } else {
    next()
  }
}
 
// For now, if the project type is not found, just redirect to project menu
export const fetchProjectType = async (to, from, next) => {
  let projectType = store.getters['listing/projectType']
  if (!projectType || parseInt(to.params.id) !== projectType.id) {
    projectType = await store.dispatch('listing/fetchProjectType', to.params.id)
  }
  if (!projectType) {
    window.location.href = '/menu'
  } else {
    next()
  }
}
 
export default {
  path: '/post/project',
  component: () => import(/* webpackChunkName: "listing-layout" */ '@/components/listing/Layout'),
  children: [
    // this URL should take project type ID as param
    {
      path: 'new/:id',
      name: 'listing:new',
      props: { newProject: true, step: 'deliverables' },
      component: components.deliverables,
      beforeEnter: fetchProjectType
    },
    // all these other URLs take project ID as param
    {
      path: ':mode/deliverables/:id',
      name: 'listing:deliverables',
      props: { step: 'deliverables' },
      component: components.deliverables,
      beforeEnter: fetchProject
    },
    {
      path: ':mode/resources/:id',
      name: 'listing:resources',
      props: { step: 'resources' },
      component: components.resources,
      beforeEnter: fetchProject
    },
    {
      path: ':mode/impact/:id',
      name: 'listing:impact',
      props: { step: 'impact' },
      component: components.impact,
      beforeEnter: fetchProject
    },
    {
      path: ':mode/image/:id',
      name: 'listing:image',
      props: { step: 'image' },
      component: components.image,
      beforeEnter: fetchProject
    },
    {
      path: ':mode/final/:id',
      name: 'listing:final',
      props: { step: 'final' },
      component: components.final,
      beforeEnter: fetchProject
    },
    {
      path: ':mode/review/:id',
      name: 'listing:review',
