export const stickyTopMx = {
  methods: {
    stickyTopMx ({el, elTop, scrollTop, stickyClass, isMobile, mobileClass, bodyOffset, hasBodyScroll, isOpen}) {
      const body = document.body
      let parentEl = this.$jq(el).parent()
      parentEl.css('z-index', isOpen ? 9999999 : 19)
      if (!el || !el.classList) {
        return false
      }
 
      if (isMobile) {
        el.classList.add(mobileClass || 'menu-fixed-mobile-top')
        if (hasBodyScroll) {
          body.classList.add('no-scroll')
        }
        if (!isOpen) {
          el.classList.remove(mobileClass || 'menu-fixed-mobile-top')
        }
      } else {
        const rect = el.getBoundingClientRect()
        if ((rect.top <= (elTop || 0)) && (window.scrollY >= (scrollTop || 200))) {
          el.classList.add(stickyClass || 'menu-fixed-top')
          body.style.paddingTop = `${bodyOffset || 0}px`
        } else {
          el.classList.remove(stickyClass || 'menu-fixed-top')
          body.style.paddingTop = '0'
        }
        if (hasBodyScroll) {
          body.classList.remove('no-scroll')
        }
      }
    }
  }
}
