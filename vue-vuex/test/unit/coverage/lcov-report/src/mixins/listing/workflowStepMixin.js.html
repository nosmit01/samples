const titles = {
  deliverables: 'Deliverables',
  resources: 'Resources',
  impact: 'Impact',
  image: 'Image',
  final: 'Final Details',
  review: 'Review'
}
 
const categories = {
  new: 'Post New Project',
  edit: 'Edit New Project',
  'edit-live': 'Edit Live Project',
  'edit-draft': 'Edit Draft Project'
}
 
export default {
  props: {
    step: {
      type: String,
      required: true,
      validation (value) {
        return [
          'deliverables',
          'resources',
          'impact',
          'image',
          'final',
          'review'
        ].indexOf(value) !== -1
      }
    }
  },
  data () {
    return {
      isChanged: false
    }
  },
  mounted () {
    this.trackStepEvent('View')
  },
  metaInfo () {
    return {
      title: `Catchafire | Post a Project | ${categories[this.mode]} | ${titles[this.step]}`
    }
  },
  computed: {
    mode () {
      return this.$route.params.mode || 'new'
    },
    stepTitle () {
      return titles[this.step]
    },
    submitting () {
      return this.$store.getters.submitting
    }
  },
  methods: {
    trackEvent (event, action, opts = {}) {
      // label consists of root | step (optional) | action
      let label = [opts.root || 'Post a Project']
 
      if (opts.step) {
        label.push(this.stepTitle)
      }
 
      if (action) {
        label.push(action)
      }
 
      this.$tracker.event(
        event,
        label.join(' | '),
        'Post a Project'
      )
    },
    trackStepEvent (event, action, opts) {
      this.trackEvent(event, action, { ...opts, step: true })
    },
    trackExit () {
      this.trackEvent('Submit', 'Save & Exit')
    },
    trackShowExample () {
      this.trackEvent('Click', 'Expand Tips Example')
    },
    trackShowMobileInfo () {
      this.trackEvent('Click', 'Show Mobile Tips')
    },
    redirect (location) {
      this.$store.dispatch('listing/redirect')
      window.location.href = location
    },
    redirectToDashboard () {
      this.redirect('/dashboard')
    }
  },
  watch: {
    formData: {
      handler () {
        this.isChanged = true
      },
      deep: true
    },
    multipartData: {
      handler () {
        this.isChanged = true
      },
      deep: true
    }
  }
}
