export const handleWorkflowStep = async (commit, fn) => {
  commit('loading', true)
  commit('clearFormErrors')
  try {
    await fn()
    commit('loading', false)
    return true
  } catch (e) {
    commit('loading', false)
    handleServerError(commit, e)
    return false
  }
}

export const mergeFormData = (data, ...others) => {
  // merge multiple FormData instances
  forEach(others, other => {
    if (other instanceof FormData) {
      for (let item of other.entries()) {
        data.append(item[0], item[1])
      }
    }
  })
  return data
}
 
export const handleServerError = (commit, err) => {
  if (err.response && err.response.status === 400 && !!err.response.data.errors) {
    const errors = flatten(map(Object.keys(err.response.data.errors), key => err.response.data.errors[key]))
    commit('formErrors', errors)
  } else if (err.response) {
    commit('formErrors', ['Sorry, an error occurred on posting your form'])
  } else {
    // not a network error, re-raise
    throw err
  }
}
