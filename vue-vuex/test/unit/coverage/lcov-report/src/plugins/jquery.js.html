// plugin to use EXISTING jquery instance on site, to prevent extra loading of another dependency.
const JQuery = {
  install (Vue, options) {
    Vue.prototype.$jq = window.hasOwnProperty('$') ? window.$ : function () {
      throw new Error('jQuery not available')
    }
  }
}
 
export default JQuery
