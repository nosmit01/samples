// Segment analytics tracker
const Tracker = {
  install (Vue, options) {
    Vue.prototype.$tracker = {
      link (element, event, label, category) {
        window.analytics && element && window.analytics.trackLink(element, `${event} | ${label}`, { label, category })
      },
      track (name, options) {
        return new Promise(async resolve => {
          if (!window.analytics) {
            console.log('Analytics not available')
            return resolve(false)
          }
          await window.analytics.track(name, options)
          resolve(true)
        })
      },
      event (event, label, category, options = {}) {
        return this.track(`${event} | ${label}`, { ...options, label, category })
      }
    }
  }
}
 
export default Tracker
