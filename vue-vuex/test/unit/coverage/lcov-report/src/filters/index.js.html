export function formatInteger (value) {
  // formats a positive whole integer, adding commas appropriately.
  const str = String(parseInt(Math.abs(Number(value) || 0)).toFixed(0))
  const pos = str.length > 3 ? str.length % 3 : 0
  return (pos ? str.substr(0, pos) + ',' : '') + str.substr(pos)
}
