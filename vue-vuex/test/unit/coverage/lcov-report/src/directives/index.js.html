/*
clickOutside Directive captures a click outside of an element:
<template>
  <div v-click-outside="doSomething"></div>
</template>
<script>
import { clickOutside } from '@/directives'
 
export default {
  directives: {
    clickOutside
  }
  methods: {
    doSomething () {
      ...
    }
  }
}
</script>
 */
export const clickOutside = {
  bind: function (el, binding, vnode) {
    el.clickOutsideEvent = function (event) {
      let target = event.target
      if (event.type === 'input:focus') target = event.target.activeElement
      // check if the click was outside el and children
      if (!(el === event.target || el.contains(target))) {
        // call method provided in attribute value
        vnode.context[binding.expression](event)
      }
    }
    document.body.addEventListener('click', el.clickOutsideEvent)
    // handle input focus since click does not register on input focus
    document.addEventListener('input:focus', el.clickOutsideEvent, true)
  },
  unbind: function (el) {
    document.body.removeEventListener('click', el.clickOutsideEvent)
    document.removeEventListener('input:focus', el.clickOutsideEvent)
  }
}
 
/*
stickyTop Directive sticks an element to the top of the screen on scroll:
 
Properties
elTop: y position of element when sticky should be applied
scrollTop: y position of scrollbar when sticky should be applied
class: class to be applied to sticky element
isMobile: whether or not screen is in mobile view
mobileClass: class to be applied to sticky element in mobile
bodyOffset: how much top padding should the body have
isOpen: boolean to track toggled states (ie. filter drop downs)
 
Implementation
<template>
  <div v-sticky-top="{elTop : 0, scrollTop : 276, class : 'fixed-top', isMobile: true}"></div>
</template>
<script>
import { stickyTop } from '@/directives'
 
export default {
  directives: {
    stickyTop
  }
}
</script>
 */
export const stickyTop = {
  bind: function (el, binding) {
    if (document.body.classList.contains('no-scroll')) {
      disableBodyScroll(el)
    }
    el.setTopPosition = function (event) {
      const body = document.body
      if (!body.classList.contains('no-scroll')) {
        enableBodyScroll(el)
      }
      if (!el || !el.classList || (binding.value.isMobile && binding.value.isOpen)) {
        return false
      }
 
      if (binding.value.isMobile) {
        el.classList.add(get(binding.value, 'mobileClass', 'menu-fixed-mobile-top'))
      } else {
        const rect = el.getBoundingClientRect()
        if ((rect.top <= get(binding.value, 'elTop', 0)) && (window.scrollY >= get(binding.value, 'scrollTop', 200))) {
          el.classList.add(get(binding.value, 'class', 'fixed-top'))
          body.style.paddingTop = `${get(binding.value, 'bodyOffset', '0')}px`
        } else {
          el.classList.remove(get(binding.value, 'class', 'fixed-top'))
          body.style.paddingTop = '0'
        }
      }
    }
    document.addEventListener('scroll', el.setTopPosition)
  },
  unbind: function (el) {
    document.removeEventListener('scroll', el.setTopPosition)
    clearAllBodyScrollLocks()
  }
}
