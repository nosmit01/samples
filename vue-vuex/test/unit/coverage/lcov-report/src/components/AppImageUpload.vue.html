<template>
<div class="image-upload-container ">
  <div class="widget-container" v-if="!selector"><input type="file" :id="widgetId"></div>
 
  <input type="hidden" :id="`${modalId}_${filenameId}`" :name="filenameName" value="">
  <input type="hidden" :id="`${modalId}_${base64Id}`" :name="base64Name" value="">
 
  <div class="modal image-upload-modal" :class="modalClass" :id="modalId" tabindex="-1" role="dialog">
    <div class="modal-dialog" role="document">
      <div class="modal-content">
 
        <div class="modal-header">
          <h5 class="modal-title">{{ title }}</h5>
          <button type="button" class="close btn-cancel" aria-label="Close" :id="cancelId">
            <span aria-hidden="true" class="ico-close-gray"></span>
          </button>
        </div>
 
        <div class="modal-body">
 
          <small v-if="header" class="modal-subtitle image-upload-helptext">{{ header }}</small>
 
          <div class="img-container" :class="containerClass" style="display: none">
            <img :id="imageId" style="max-width: 100%;" />
          </div>
 
          <div class="cr-slider-wrap" style="display: none">
            <input class="cr-slider" type="range" step="0.0001" aria-label="zoom" min="0.01" max="3.5">
          </div>
 
          <div class="empty-container mx-auto" :class="containerClass">
            <div class="errors-container" style="display:none;"></div>
            <input v-if="!!selector" type="file" :id="widgetId">
          </div>
 
          <small v-if="footer" class="modal-note image-upload-helptext">{{ footer }}</small>
 
        </div>
 
        <!-- only show once file selected -->
        <div class="modal-footer">
          <div class="modal-buttons" style="display:none;">
            <button type="button" class="btn btn-primary btn-save" :id="applyId">Apply</button>
          </div>
        </div>
      </div>
    </div>
  </div>
</div>
</template>
 
<script>
export default {
  props: {
    name: {
      type: String,
      required: true
    },
    type: {
      validator (value) {
        return ['cover', 'square', 'circle'].indexOf(value) !== -1
      }
    },
    title: {
      type: String,
      default: 'Upload Your Image'
    },
    header: {
      type: String,
      required: false
    },
    footer: {
      type: String,
      required: false
    },
    label: {
      type: String,
      required: false
    },
    selector: {
      type: String,
      required: false
    },
    cropWidth: {
      type: Number,
      default: 175
    },
    cropHeight: {
      type: Number,
      default: 175
    },
    viewportWidth: {
      type: Number,
      default: 330
    },
    viewportHeight: {
      type: Number,
      default: 330
    },
    allowCrop: {
      type: Boolean,
      default: true
    },
    maxSize: {
      type: Number,
      default: 5120
    }
  },
  data () {
    const circle = this.type === 'circle'
    const cover = this.type === 'cover'
    const containerClass = cover ? 'cover-image' : 'square'
    const widgetId = `id_${this.name}`
    const modalId = `image-upload-modal-${widgetId}`
    const imageId = `${modalId}_image`
    const filenameId = `image-upload-filename-${widgetId}`
    const base64Id = `image-upload-base64-${widgetId}`
    const cancelId = `${modalId}_cancel`
    const applyId = `${modalId}_apply`
    const filenameName = `${this.name}_filename`
    const base64Name = `${this.name}_base64`
 
    let modalClass = ''
 
    if (cover) {
      modalClass = 'image-modal-no-viewport-border'
    } else if (circle) {
      modalClass = 'image-upload-modal-circle'
    }
 
    return {
      circle,
      cover,
      modalClass,
      containerClass,
      modalId,
      imageId,
      widgetId,
      filenameId,
      base64Id,
      cancelId,
      applyId,
      filenameName,
      base64Name
    }
  },
  mounted () {
    const $el = this.$jq(`#${this.widgetId}`)
    // requires we have image upload JS libraries globally available
    $el.imageUpload({
      label: this.label,
      circle: this.circle,
      selector: this.selector,
      modal: this.modalId,
      base64Input: this.base64Id,
      filenameInput: this.filenameId,
      cropWidth: this.cropWidth,
      cropHeight: this.cropHeight,
      viewportWidth: this.viewportWidth,
      viewportHeight: this.viewportHeight,
      allowCrop: this.allowCrop,
      maxSize: this.maxSize
    })
    // "data" is a FormData object consisting of filename_{name} and base64_{name} as keys
    $el.on('imageupload:save', (event, url, data) => this.$emit('save', url, data))
    $el.on('imageupload:cancel', () => this.$emit('cancel'))
    $el.on('imageupload:error', msg => this.$emit('error', msg))
  }
}
</script>
