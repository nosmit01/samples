<template>
  <div class="caf-carousel" :style="styles.carousel">
    <!-- Desktop -->
    <slick
      v-if="!isMobile"
      class="row"
      :class="{'alignToGrid': alignToGrid}"
      ref="slick"
      @afterChange="$emit('navigate')"
      :options="slickOptions"
    >
      <div
        class="item"
        v-for="item in items"
        :key="item[carouselItemKeyProp]"
        :class="carouselItemClass"
        @click="$emit('carousel-item-click', item)"
      >
        <slot :item="item"></slot>
      </div>
    </slick>
 
    <!-- Mobile -->
    <div
      v-if="isMobile"
      class="caf-slider"
      :style="styles.slider"
      @scroll="$emit('swipe')"
    >
      <div class="row list">
        <div
          class="item"
          v-for="item in items"
          :key="item[carouselItemKeyProp]"
          :class="carouselItemClass"
          @click="$emit('carousel-item-click', item)"
        >
          <slot :item="item"></slot>
        </div>
      </div>
    </div>
  </div>
</template>
 
<script>
import Vue from 'vue'
import { createNamespacedHelpers } from 'vuex'
import Slick from 'vue-slick'
 
const { mapGetters } = createNamespacedHelpers('breakpoints')
 
export default {
  props: {
    items: {
      type: Array,
      required: true
    },
    carouselItemKeyProp: {
      type: String,
      default: 'id'
    },
    carouselItemClass: {
      type: String,
      required: false
    },
    alignToGrid: {
      type: Boolean,
      default: false
    },
    visible: {
      type: Number,
      default: 3
    },
    mobileVisible: {
      type: Number,
      default: 4
    },
    cycle: {
      type: Number,
      default: 3
    }
  },
  components: {
    Slick
  },
  data () {
    return {
      slickOptions: {
        slidesToShow: this.visible,
        nextArrow: `<div id="caf-carousel-arrowR" :style="styles.arrows">
        <div class="arrow">
        <span class="ico ico-arrow-right-small"></span>
        </div>
        </div>`,
        prevArrow: `<div id="caf-carousel-arrowL" :style="styles.arrows">
        <div class="arrow">
        <span class="ico ico-arrow-left-small"></span>
        </div>
        </div>`
      },
      moving: false,
      index: 0,
      styles: {},
      slideTransition: 'slide-from-right'
    }
  },
  mounted () {
    this.configureStyles()
  },
  computed: {
    ...mapGetters(['isMobile', 'isDesktop'])
  },
  watch: {
    isMobile: 'configureStyles',
    items: 'configureStyles'
  },
  methods: {
    configureStyles () {
      // make sure we're mounted
      if (!this.$el) {
        return
      }
      // wait for update to DOM first
      // as we need to do calculations based on current DOM state
      Vue.nextTick(() => {
        this.styles = this.isDesktop ? this.getDesktopStyles() : this.getMobileStyles()
      })
    },
    getDesktopStyles () {
      this.$refs.slick.reSlick()
      const $items = this.$el.querySelector('.item')
      const $arrowL = this.$el.querySelector('#caf-carousel-arrowL')
      const $arrowR = this.$el.querySelector('#caf-carousel-arrowR')
 
      if ($arrowL) {
        $arrowL.style.height = `${$items.offsetHeight - 18}px`
      }
      if ($arrowR) {
        $arrowR.style.height = `${$items.offsetHeight - 18}px`
      }
 
      return {}
    },
    getMobileStyles () {
      const $items = this.$el.querySelector('.item')
      const styles = {}
 
      styles.carousel = {
        width: '100%',
        position: 'relative',
        left: 0
      }
 
      styles.slider = {
        width: `${$items.offsetWidth * this.items.length}px`
      }
 
      return styles
    },
    next () {
      this.$refs.slick.next()
    },
    prev () {
      this.$refs.slick.prev()
    }
  }
}
</script>
