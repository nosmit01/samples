<template>
  <app-carousel :items="volunteers" carousel-item-class="caf-card-volunteer" @carousel-item-click="jumpTo">
    <template slot-scope="itemProps">
      <!--<div class="ico-star-hollow"></div>--> <!-- use class="ico-star-filled" if selected -->
      <div class="front">
        <div class="caf-card-volunteer-header">
          <div class="offset-container d-flex justify-content-between align-items-center">
            <span class="profile-image">
              <img v-if="itemProps.item.image" :src="itemProps.item.image" />
              <img v-else class="" width="" src="~__STATIC__/images/responsive/placeholders/placeholder_user_profile_blank.svg" />
            </span>
            <span class="affiliate-logo" v-if="itemProps.item.badge">
              <img :src="itemProps.item.badge" :alt="itemProps.item.employer">
            </span>
          </div>
        </div>
        <div class="caf-card-volunteer-body pt-2 pb-3 px-3">
          <h4 class="caf-card-volunteer-body-title my-0">
            {{ itemProps.item.full_name }}.
          </h4>
          <p class="caf-card-volunteer-body-subtitle my-0 text-color-emperor" v-if="itemProps.item.city && itemProps.item.country">
          {{ itemProps.item.city }}, {{ itemProps.item.state || itemProps.item.country }}
          </p>
          <div class="caf-card-volunteer-body-main mt-3">
            <div>
              <span class="badge badge-default" v-for="skill in itemProps.item.skills" :key="skill">{{ skill }}</span><br>
            </div>
          </div>
          <p class="caf-card-volunteer-body-footer mt-3 mb-0 text-color-emperor">
          {{ itemProps.item.first_name }}
          <span v-if="itemProps.item.tidbit === 'projects_complete'">
            has <strong>completed over {{ itemProps.item.num_projects_complete }} successful projects</strong> with other organizations.
          </span>
          <span v-if="itemProps.item.tidbit === 'experience'">
            has <strong>10 or more years experience</strong> in their field.
          </span>
          <span v-if="itemProps.item.tidbit === 'project_interest'">
            has <strong>shown interest</strong> in this specific type of project.
          </span>
          <span v-if="itemProps.item.tidbit === 'high_nps'">
            is <strong>recommended highly</strong> by other organizations.
          </span>
          <span v-if="itemProps.item.tidbit === 'matching_cause'">
            is <strong>passionate about {{ itemProps.item.matching_cause }}</strong>.
          </span>
          <span v-if="!itemProps.item.tidbit">
            <strong>is active and ready</strong> to volunteer.
          </span>
          </p>
        </div>
      </div>
      <div class="back text-color-white px-3 mb-3">
        <h4 class="pt-4">
          {{ itemProps.item.full_name }}.
        </h4>
        <p class="subtitle mt-2 mb-0" v-if="itemProps.item.position">
        {{ itemProps.item.title }} <span v-if="itemProps.item.position"> at {{ itemProps.item.position }}</span>
        </p>
        <p class="bio mt-1 mb-0">
        {{ itemProps.item.bio|truncateWords(50) }}
        </p>
      </div>
    </template>
  </app-carousel>
</template>
 
<script>
import AppCarousel from '@/components/AppCarousel'
 
export default {
  props: {
    volunteers: {
      type: Array,
      required: true
    }
  },
  filters: {
    // tbd: move this into another package
    truncateWords (value, length) {
      if (!value) return ''
      // look for first sentence
      const sentence = value.split('.')[0]
      const words = sentence.split(' ')
      if (words.length > length) {
        return words.slice(0, length).join(' ') + ' ...'
      }
      return sentence + '.'
    }
  },
  components: {
    AppCarousel
  },
  methods: {
    jumpTo (item) {
      if (item.url) {
        window.document.location.href = item.url
      }
    }
  }
}
</script>
