<template>
  <div>
    <nav-header />
 
    <filter-bar
      :is-wishlist="true"
    />
 
    <div
      v-if="!loading && pageObj.count === 0"
      class="caf-card-wishlist-illustration"
    >
      <span class="py-2">You don’t have anything saved yet</span>
      <div class="divider"></div>
      <h6 class="h5-md mt-4 mb-3 mb-md-2 font-weight-normal text-center">
        Keep track of projects you want to do in the future by saving them to your Wishlist
      </h6>
      <img :src="cardIllustration" class="card-ill">
    </div>
 
    <div v-if="!loading && pageObj.count > 0" class="container hidden-md-up">
      <app-tab
        :init="init"
        :tabs="tabs"
        :activeTab="activeTab"
        @tab-change="updateTabs"
        @clear-init="init = false"
        :tracking="{label: 'SF Menu Show Wishlist Tab', category: 'Storefront Menu'}"
      />
    </div>
 
    <project-card-grid-header
      class="mb-3 mb-md-0 pt-1 pt-md-0"
      :categories="categories"
      :sortOptions="sortOptions"
      :showSortOptions="pageObj.count > 1"
      :defaultSortOption="defaultSortOption"
      @sort-option-change="reorder"
      v-if="pageObj && pageObj.count > 0"
      >
      <div v-if="loading">
        Loading projects...
      </div>
      <div v-if="!loading && pageObj.count > 0" class="hidden-sm-down">
        <app-tab
          :init="init"
          :tabs="tabs"
          :activeTab="activeTab"
          @tab-change="updateTabs"
          @clear-init="init = false"
          :tracking="{label: 'SF Menu Show Wishlist Tab', category: 'Storefront Menu'}"
        />
      </div>
    </project-card-grid-header>
 
    <div v-if="!loading && pageObj.count > 0" class="divider opacity-1 mb-3 hidden-sm-down"></div>
 
    <app-grid-list
      v-if="!loading && pageObj.count > 0"
      :cards="tabbedProjects"
      :loading="loading"
      :page="pageObj"
      :usePagination="false"
      @list-view="listView"
    >
      <template slot="grid">
        <div class="col-md-4" v-for="project in tabbedProjects" :key="project.id">
          <project-card
            :project="project"
            :tracking="{label: 'SF Menu Project Cards', category: 'Storefront Menu'}"
            class="mb-3"
          ></project-card>
        </div>
      </template>
 
      <template slot="list-header">
        <div class="width-1 px-1">Save</div>
        <div class="width-6 px-1">Project</div>
        <div class="width-2 px-1">Estimated Time</div>
        <div class="width-3 px-1">Category</div>
      </template>
 
      <template slot="list">
        <project-row
          v-for="project in tabbedProjects" :key="project.id"
          :project="project"
          :tracking="{label: 'SF Menu Project List Row', category: 'Storefront Menu'}"
        ></project-row>
      </template>
 
    </app-grid-list>
  </div>
</template>
 
<script>
import { createNamespacedHelpers } from 'vuex'
 
import NavHeader from '@/components/storefront/NavHeader'
import FilterBar from '@/components/storefront/FilterBar'
import AppGridList from '@/components/AppGridList'
import AppTab from '@/components/AppTab'
import ProjectCard from '@/components/storefront/ProjectCard'
import ProjectRow from '@/components/storefront/ProjectRow'
import ProjectCardGridHeader from '@/components/storefront/ProjectCardGridHeader'
import { pluralize } from '@/filters'
import filter from 'lodash/filter'
import map from 'lodash/map'
import find from 'lodash/find'
import get from 'lodash/get'
import cardIllustration from '__STATIC__/images/responsive/projects/card-illustration.svg'
 
const { mapGetters, mapActions } = createNamespacedHelpers('storefront')
const { mapGetters: mapBreakpointGetters } = createNamespacedHelpers('breakpoints')
 
export default {
  components: {
    NavHeader,
    FilterBar,
    AppGridList,
    AppTab,
    ProjectCard,
    ProjectRow,
    ProjectCardGridHeader
  },
  metaInfo: {
    title: 'Starred'
  },
  filters: {
    pluralize
  },
  computed: {
    tabbedProjects () {
      return filter(this.projects, { type: this.activeTab })
    },
    ...mapGetters([
      'categories',
      'pageObj',
      'projects',
      'wishlist',
      'loading',
      'wishlistEnabled',
      'isWishlisted',
      'isListView'
    ]),
    ...mapBreakpointGetters(['isDesktop', 'isMobile'])
  },
  data () {
    return {
      cardIllustration,
      init: true,
      sortOptions: [
        {
          value: 'name',
          label: 'Name (A-Z)'
        },
        {
          value: 'recent',
          label: 'Recently Added'
        }
      ],
      defaultSortOption: this.$route.query.order || 'recent',
      tabs: [
        {
          id: 'project',
          label: 'Projects'
        },
        {
          id: 'call',
          label: 'Calls'
        }
      ],
      activeTab: ''
    }
  },
  beforeRouteEnter (to, from, next) {
    next(vm => {
      vm.setQueryString()
      next()
    })
  },
  methods: {
    setQueryString () {
      if (this.$route.query.view) {
        this.setListView(true)
      }
      this.$router.replace({
        query: Object.assign({}, this.$route.query, {
          view: this.isListView ? 'list' : undefined
        })
      })
    },
    doFetch () {
      this.fetchWishlist({...this.$route.query, limit: 1000})
    },
    reorder (order) {
      const query = {...this.$route.query, order}
      this.$router.push({
        name: this.$route.name,
        replace: true,
        query
      })
    },
    ...mapActions(['fetchWishlist', 'setListView']),
    listView () {
      this.setListView(true)
      this.$router.push({
        name: this.$route.name,
        replace: true,
        query: Object.assign({}, this.$route.query, {
          ns: 1,
          view: 'list'
        })
      })
    },
    updateTabs (id) {
      let tabs = [...this.tabs]
 
      // set tab counts
      tabs = map(tabs, tab =>
        Object.assign({}, tab, {
          count: filter(this.projects, project =>
            project.type === tab.id && this.isWishlisted(project.id)).length
        }))
 
      // set active and disabled tabs
      if (this.init) {
        // set to first available tab with count > 0
        this.activeTab = get(find(tabs, tab => tab.count > 0), 'id')
      } else {
        this.activeTab = id || this.activeTab
      }
 
      this.tabs = map(tabs, tab =>
        Object.assign({}, tab, {
          active: tab.id === this.activeTab,
          // make sure tab is not active tab, or else leave active
          disabled: tab.id !== this.activeTab && tab.count === 0
        }))
    }
  },
  watch: {
    // route changes are fired from nav toolbar or in the URL. We just pick up the changes here and update the store.
    $route: {
      handler: 'doFetch',
      immediate: true
    },
    // Purposely using 'projects' and 'wishlist' to update tab counts.
    // 'project' is used once to get and init all projects in a wishlist.
    // 'wishlist' is used every time wishlist gets updated so that we can change the count, but keep
    // the unchecked cards on the page in case the user wants to re-check them.
    projects () {
      this.updateTabs()
    },
    wishlist () {
      this.updateTabs(this.activeTab)
    }
  }
}
</script>
