<template>
  <div v-if="isFeaturedProjects && !loadingFeatured" class="container mt-3 mt-md-5 storefront-featured-projects hidden-print">
    <div class="row">
      <div class="col-md-12">
        <ul class="storefront-featured-projects-menu d-flex justify-content-start align-items-center mb-md-2 mb-1 ml-1 ml-md-0 text-color-emperor">
          <li class="font-weight-semibold" :class="activeTabClass('recommended')" @click="showRecommendedTab">
            Recommended
          </li>
          <li class="font-weight-semibold ml-4" :class="activeTabClass('new')" @click="showNewTab">
            New
          </li>
          <li class="font-weight-semibold ml-4" :class="activeTabClass('bundles')" @click="showBundlesTab" v-if="isFeaturedBundles">
            Bundles
            <!-- remove this after a month -->
            <span v-show="showBundlesDot" class="ico-new-feature" style="top: -2px; right: -12px;"></span>
          </li>
          <li class="ml-auto hidden-sm-down font-weight-semibold see-all-link" @click="seeAll">
            See All {{ seeAllLabel }}
          </li>
        </ul>
      </div>
    </div>
 
    <div class="storefront-featured-projects-projects row hidden-sm-down" v-if="showBundles">
      <div class="col-md-6">
        <bundle-card
          class="mb-0"
          size="lg"
          :bundle="featuredBundles[0]"
          :key="featuredBundles[0].id"
          :tracking="tracking.heroCard"
        ></bundle-card>
      </div>
      <div class="col-md-6">
        <bundle-card
          v-for="bundle in featuredBundles.slice(1, 4)"
          :key="bundle.id"
          :bundle="bundle"
          :tracking="tracking.smallCard"
          size="sm"
        ></bundle-card>
      </div>
    </div>
    <bundle-card-carousel
      v-if="showBundles"
      class="hidden-md-up"
      :bundles="featuredBundles"
      :carousel-tracking="tracking.smallCard"
      :card-tracking="tracking.smallCard"
    />
 
    <div class="storefront-featured-projects-projects row hidden-sm-down" v-if="showRecommendedProjects">
      <div class="col-md-6">
        <project-card
          class="mb-0"
          size="lg"
          :project="featuredRecommendedProjects[0]"
          :key="featuredRecommendedProjects[0].id"
          :tracking="tracking.heroCard"
        ></project-card>
      </div>
 
      <div class="col-md-6">
        <project-card
           v-for="project in featuredRecommendedProjects.slice(1, 4)"
           :key="project.id"
          :project="project"
          :tracking="tracking.smallCard"
          size="sm"
        ></project-card>
      </div>
    </div>
 
    <project-card-carousel
      v-if="showRecommendedProjects"
      class="hidden-md-up"
      :projects="featuredRecommendedProjects"
      :carousel-tracking="tracking.smallCard"
      :card-tracking="tracking.smallCard"
    />
 
    <div class="storefront-featured-projects-projects row hidden-sm-down" v-if="showNewProjects">
      <div class="col-md-6">
        <project-card
          class="mb-0"
          size="lg"
          :project="featuredNewProjects[0]"
          :key="featuredNewProjects[0].id"
          :tracking="tracking.heroCard"
        ></project-card>
      </div>
 
      <div class="col-md-6">
        <project-card
          v-for="project in featuredNewProjects.slice(1, 4)"
          :key="project.id"
          :project="project"
          :tracking="tracking.smallCard"
          size="sm"
        ></project-card>
      </div>
    </div>
 
    <project-card-carousel
      v-if="showNewProjects"
      class="hidden-md-up"
      :projects="featuredNewProjects"
      :carousel-tracking="tracking.smallCard"
      :card-tracking="tracking.smallCard"
    />
 
    <div class="hidden-md-up text-color-emperor text-center mb-2" @click="seeAll">See All</div>
  </div>
</template>
 
<script>
import { createNamespacedHelpers } from 'vuex'
 
import BundleCard from '@/components/storefront/BundleCard'
import BundleCardCarousel from '@/components/storefront/BundleCardCarousel'
import ProjectCard from '@/components/storefront/ProjectCard'
import ProjectCardCarousel from '@/components/storefront/ProjectCardCarousel'
 
const { mapGetters, mapActions } = createNamespacedHelpers('storefront')
const { mapGetters: mapBreakpointGetters } = createNamespacedHelpers('breakpoints')
 
export default {
  components: {
    BundleCard,
    BundleCardCarousel,
    ProjectCard,
    ProjectCardCarousel
  },
  props: {
    resetHome: {
      type: Boolean,
      default () {
        return false
      }
    },
    initialTab: {
      type: String,
      default: 'recommended'
    }
  },
  created () {
    this.fetchProjects('popular')
    this.fetchProjects('recent')
    this.fetchFeaturedBundles()
  },
  data () {
    return {
      tab: this.initialTab,
      tabLabel: {
        recommended: 'Recommended',
        new: 'New',
        bundles: 'Bundles'
      },
      showBundlesDot: !localStorage.getItem('showBundlesDot')
    }
  },
  computed: {
    seeAllLabel () {
      return this.tabLabel[this.tab]
    },
    isFeaturedProjects () {
      return this.featuredRecommendedProjects.length > 3
    },
    isFeaturedBundles () {
      return this.featuredBundles.length > 3
    },
    showRecommendedProjects () {
      return this.tab === 'recommended' && this.isFeaturedProjects
    },
    showNewProjects () {
      return this.tab === 'new' && this.isFeaturedProjects
    },
    showBundles () {
      return this.tab === 'bundles' && this.isFeaturedBundles
    },
    tracking () {
      const category = 'Featured'
      const label = `Featured: ${this.seeAllLabel} `
      return {
        heroCard: {label: label + ' Hero Card', category},
        smallCard: {label: label + ' Small Card', category}
      }
    },
    ...mapGetters([
      'loadingFeatured',
      'featuredBundles',
      'featuredRecommendedProjects',
      'featuredNewProjects',
      'isListView'
    ]),
    ...mapBreakpointGetters(['isDesktop', 'isMobile'])
  },
  methods: {
    fetchProjects (order = 'popular') {
      this.fetchFeaturedProjects({
        order,
        isFeatured: true
      })
    },
    activeTabClass (tab) {
      return {active: this.tab === tab}
    },
    showRecommendedTab () {
      this.tab = 'recommended'
      this.$tracker.event('Click', 'Featured: Recommended Tab', 'Featured')
    },
    showNewTab () {
      this.tab = 'new'
      this.$tracker.event('Click', 'Featured: New Projects Tab', 'Featured')
    },
    showBundlesTab () {
      this.tab = 'bundles'
      this.$tracker.event('Click', 'Featured: Bundles Tab', 'Featured')
      // TBD: remove this after a month: no longer required
      this.showBundlesDot = false
      localStorage.setItem('showBundlesDot', true)
    },
    seeAll () {
      this.$tracker.event('Click', `Featured: ${this.tabLabel[this.tab]} See All`, 'Featured')
      this.$router.push({
        name: `storefront:${this.tab}`,
        replace: true,
        query: {view: this.isListView ? 'list' : undefined}
      })
    },
    ...mapActions([
      'fetchFeaturedProjects',
      'fetchFeaturedBundles'
    ])
  },
  watch: {
    resetHome () {
      this.showRecommendedTab()
    }
  }
}
</script>
