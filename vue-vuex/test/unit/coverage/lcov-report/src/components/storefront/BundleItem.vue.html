<template>
  <div :class="item.type === 'project' ? '' : 'd-md-flex align-items-center h-100'">
    <project-card
      v-if="item.type === 'project'"
      :project="item.project"
      :tracking="{label: trackingLabel, category: 'Bundle Details'}"
      :bundle-item="item"
      ></project-card>
 
    <div v-else class="font-main font-weight-normal">
      <div v-if="item.header" class="text-uppercase mb-2 text-color-emperor"><u>{{ item.header }}</u></div>
      <h4 v-if="item.subheader" class="">{{ item.subheader }}</h4>
      <div
        v-if="item.body"
        v-html="markdown(item.body, true)"
        class="text-color-emperor lead"
        :class="{'font-size-16': isMobile}"
      ></div>
    </div>
  </div>
</template>
 
<script>
import { markdown } from '@/filters'
import ProjectCard from '@/components/storefront/ProjectCard'
 
export default {
  components: {
    ProjectCard
  },
  methods: {
    markdown
  },
  props: {
    item: {
      type: Object,
      required: true
    },
    isMobile: {
      type: Boolean,
      default: false
    }
  },
  data () {
    let trackingLabel
    if (this.item.project) {
      trackingLabel = `Bundle Detail ${this.item.project.type === 'call' ? 'Call' : 'Project'}`
    }
    return {
      trackingLabel
    }
  }
}
</script>
