<template>
  <div>
    <filter-bar
      v-if="!loading"
      :bundle-detail="bundle"
      :bundle-not-found="notFound"
    />
 
    <div class="container-fluid p-0 bg-caf-gray" v-if="notFound">
      <div class="row">
        <div class="col-12">
          <div class="project-listing-no-results p-3 mx-1 mx-md-0">
            <div class="text-center">
              Sorry, this project bundle is currently unavailable.
            </div>
          </div>
        </div>
      </div>
    </div>
 
    <div class="container-fluid p-0 p-md-1 loading-animation" v-if="loading">
      <div class="row storefront-mobile-margins">
        <div class="col-md-4" v-for="n in isMobile ? 3 : 9" :key="n">
          <app-loading-card></app-loading-card>
        </div>
      </div>
    </div>
 
    <div v-if="bundle">
 
      <header class="bg-color-emperor pt-2 pb-4 storefront-nav-header font-weight-semibold" :style="headerStyle">
        <div class="container d-flex flex-column align-items-center justify-content-center h-100 px-3 text-color-white">
 
          <div class="icon pt-2 pb-2">
            <img :src="stackIcon" height="30" width="30">
          </div>
 
          <div v-if="numCalls > 0 || numProjects > 0" class="pb-2 text-uppercase">
            <span v-if="numCalls > 0">{{ numCalls }} Call{{ numCalls|pluralize }}</span>
            <span v-if="numCalls > 0 && numProjects > 0"> | </span>
            <span v-if="numProjects > 0">{{ numProjects }} Project{{ numProjects|pluralize }}</span>
          </div>
          <h3 class="h2-md pb-2 text-center">
            {{ bundleName }}
          </h3>
          <h5 v-if="bundle.timeframe">
            <span class="text-capitalize">{{ bundle.timeframe }}</span> to Complete
          </h5>
        </div>
      </header>
 
      <div v-if="bundle.description" class="bg-color-white py-4 justify-content-center">
        <div class="container text-color-emperor font-weight-normal text-center" :class="isMobile ? '' : 'w-50 lead'">{{ bundle.description }}</div>
      </div>
 
      <div class="bg-color-gray-lighter">
 
        <div v-if="isMobile" class="container pt-4">
          <div class="row">
            <div class="col-12 col-md-6 mb-4" :key="item.id" v-for="item in bundle.vertical_items">
              <bundle-item :item="item" :is-mobile="isMobile" class="mobile-bundle-item"></bundle-item>
            </div>
          </div>
        </div>
 
        <div v-if="!isMobile" class="container pt-5 pb-1">
          <div class="row" :key="rowIndex" v-for="(row, rowIndex) in bundle.items">
            <div class="mt-1" :class="itemClass(item, index, row)" :key="item.id" v-for="(item, index) in row">
              <bundle-item :item="item"></bundle-item>
            </div>
          </div>
        </div>
 
      </div>
 
      <div class="bg-color-emperor justify-content-center text-color-white py-4">
 
        <div class="container">
          <div class="row">
 
            <div class="col-12 col-md-8 border-right" v-if="!!bundle.outcome.header || !!bundle.outcome.body">
              <h5 v-if="bundle.outcome.header" class="h4 mb-2"><span class="font-weight-semibold">What You'll Get:</span> {{ bundle.outcome.header }}</h5>
              <div v-if="bundle.outcome.body" v-html="markdown(bundle.outcome.body)"></div>
            </div>
 
            <div class="col-12 col-md-4 pl-5">
 
              <hr class="d-md-none bg-color-white">
 
              <div class="row">
 
                <div class="col-6 col-md-12">
 
                  <div class="d-flex d-flex-column font-size-14">
                    <div class="font-thin">MONEY SAVED</div>
                    <div class="icon pl-1">
                      <span v-tooltip="{content: tooltips.savings}">
                        <i class="fa fa-info-circle" aria-label="Info"></i>
                      </span>
                    </div>
                  </div>
 
                  <div class="mb-4">
                    <h4>${{ bundle.outcome.savings|formatInteger }}</h4>
                  </div>
                </div>
 
                <div class="col-6 col-md-12">
                  <div v-if="bundle.timeframe" class="font-size-14">
                    <span class="font-thin">TIME TO COMPLETE</span>
                  </div>
 
                  <div v-if="bundle.timeframe">
                    <h4>{{ bundle.timeframe }}</h4>
                  </div>
                </div>
 
              </div>
 
            </div>
          </div>
        </div>
      </div>
    </div>
  </div>
</template>
 
<script>
import filter from 'lodash/filter'
import map from 'lodash/map'
import flatten from 'lodash/flatten'
 
import stackIcon from '__STATIC__/images/responsive/icons/ico_stack.svg'
 
import { createNamespacedHelpers } from 'vuex'
 
import { tooltip } from '@/directives'
import { formatInteger, markdown, pluralize } from '@/filters'
 
import FilterBar from '@/components/storefront/FilterBar'
import BundleItem from '@/components/storefront/BundleItem'
import FontAwesomeIcon from '@fortawesome/vue-fontawesome'
import faArrow from '@fortawesome/fontawesome-free-solid/faArrowLeft'
// we should use bundle specific loader or a generic detail loader
import AppLoadingCard from '@/components/AppLoadingCard'
 
const { mapGetters, mapActions } = createNamespacedHelpers('storefront')
const { mapGetters: mapBreakpointGetters } = createNamespacedHelpers('breakpoints')
 
export default {
  components: {
    FilterBar,
    BundleItem,
    FontAwesomeIcon,
    AppLoadingCard
  },
  directives: {
    tooltip
  },
  filters: {
    pluralize,
    formatInteger
  },
  metaInfo () {
    return {
      title: (this.bundle && this.bundle.title) || 'Project Menu'
    }
  },
  data () {
    return {
      faArrow,
      stackIcon,
      tooltips: {
        savings: 'This is the value we\'ve estimated for all projects in this bundle. ' +
        '<a href="https://medium.com/@catchafirenyc/evolving-how-we-calculate-catchafires-project-values-8eaa200b906" ' +
        'rel="noopener" target="_blank">Learn more</a> about how we calculate project savings.'
      }
    }
  },
  methods: {
    markdown,
    getBundle () {
      this.fetchBundle(this.$route.params.id)
    },
    bundleUpdated () {
      this.$tracker.event(
        'View',
        'Bundle Details Page',
        'Bundle Details'
      )
    },
    itemClass (item, index, row) {
      if (item.type === 'project') {
        return 'col-md-4 mb-5 '
      }
      switch (row.length) {
        case 1:
          return 'col-md-12 mb-4'
        case 2:
          return index === 0 ? 'col-md-8 mb-5' : 'col-md-6 mb-5'
        default:
          return 'col-md-4 mb-5 '
      }
    },
    ...mapActions([
      'fetchBundle'
    ])
  },
  computed: {
    numCalls () {
      return filter(this.projects, project => project.type === 'call').length
    },
    numProjects () {
      return filter(this.projects, project => project.type === 'project').length
    },
    projects () {
      if (!this.bundle) {
        return []
      }
      return map(filter(flatten(this.bundle.items), item => !!item.project), item => item.project)
    },
    bundleName () {
      return (this.bundle && (this.bundle.title || this.bundle.name)) || ''
    },
    headerStyle () {
      if (this.bundle.image) {
        return {
          background: `url(${this.bundle.image}) no-repeat center center`,
          backgroundSize: 'cover'
        }
      }
      return { background: 'none' }
    },
    ...mapBreakpointGetters(['isMobile']),
    ...mapGetters({
      loading: 'loading',
      notFound: 'bundleNotFound',
      bundle: 'bundleDetail'
    })
  },
  watch: {
    $route: {
      handler: 'getBundle',
      immediate: true
    },
    bundle: {
      handler: 'bundleUpdated'
    }
  }
}
</script>
