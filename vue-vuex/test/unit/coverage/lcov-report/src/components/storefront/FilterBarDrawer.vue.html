<template>
  <div
    v-if="isOpen"
    class="caf-filter-bar-menu"
  >
    <div class="container p-0 p-md-1">
      <div class="col-12 px-0">
        <slot></slot>
      </div>
    </div>
  </div>
</template>
 
<script>
export default {
  props: {
    isOpen: {
      type: Boolean,
      default () {
        return false
      }
    },
    isMobile: {
      type: Boolean,
      default () {
        return false
      }
    }
  },
  mounted () {
    window.addEventListener('scroll', this.setTopPosition)
  },
  beforeDestroy () {
    window.removeEventListener('scroll', this.setTopPosition)
  },
  watch: {
    isOpen () {
      setTimeout(() => {
        this.setTopPosition()
      }, 100)
    }
  },
  methods: {
    setTopPosition () {
      const el = this.$el
      if (!el || !el.classList) {
        return false
      }
 
      if (this.isMobile) {
        el.classList.add('menu-fixed-mobile-top')
      } else {
        const rect = el.getBoundingClientRect()
        if ((rect.top <= 48) && (window.scrollY >= 346)) {
          el.classList.add('menu-fixed-top')
        } else {
          el.classList.remove('menu-fixed-top')
        }
      }
    }
  }
}
</script>
