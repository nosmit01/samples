<template>
  <div
    class="caf-card-bundle"
    :class="{'carousel-card': isCarousel, large: size === 'lg', small: size === 'sm'}"
    >
    <div class="caf-card-bundle-shadow"></div>
    <div class="caf-card-bundle-container"
         @mouseleave="stopTimer"
         @mouseover="trackHover"
    >
    <!-- <div :class="wishlisted ? 'ico-star-filled' : 'ico-star-hollow'" @click="toggleWishlist" v-if="wishlistEnabled"></div>-->
      <router-link
        class="caf-card-bundle-main"
        @click.native="trackClick"
        :to="{name: 'storefront:bundle', params: {id: bundle.id, slug: bundle.slug}}"
      >
        <div class="caf-card-bundle-header mb-0" :style="styles.header"></div>
        <div class="caf-card-bundle-body pb-3 px-3">
          <h4 class="caf-card-bundle-body-title pt-3 mb-2 d-flex align-items-center">
            {{ bundle.title }}
          </h4>
          <div class="caf-card-bundle-body-subtitle mb-3 text-color-emperor" v-if="bundle.num_calls || bundle.num_projects">
            <span class="ico-stack-dark mr-1"></span>
            <span v-if="bundle.num_calls">{{ bundle.num_calls }} Call{{ bundle.num_calls|pluralize }}</span>
            <span v-if="bundle.num_calls && bundle.num_projects">&nbsp;|&nbsp;</span>
            <span v-if="bundle.num_projects">{{ bundle.num_projects }} Project{{ bundle.num_projects|pluralize }}</span>
          </div>
          <div class="caf-card-bundle-body-description text-color-emperor">
            {{ bundle.description }}
          </div>
        </div>
      </router-link>
    </div>
  </div>
</template>
 
<script>
import { pluralize } from '@/filters'
 
export default {
  props: {
    bundle: {
      type: Object,
      required: true
    },
    size: {
      type: String, // lg or sm
      required: false
    },
    isCarousel: {
      type: Boolean,
      default: false
    },
    tracking: {
      type: Object,
      required: false
    }
  },
  filters: {
    pluralize
  },
  methods: {
    trackHover () {
      if (this.tracking && !this.timer && !this.trackHoverComplete) {
        this.timer = window.setTimeout(() => {
          // double check in case of race condition
          if (!this.trackHoverComplete) {
            this.$tracker.event('Hover', this.tracking.label, this.tracking.category)
          }
          this.trackHoverComplete = true
          this.stopTimer()
        }, 1500)
      }
    },
    stopTimer () {
      if (this.tracking) {
        this.timer && window.clearTimeout(this.timer)
        this.timer = null
      }
    },
    trackClick () {
      this.$tracker.event('Click', this.tracking.label, this.tracking.category)
    }
  },
  data () {
    const { image } = this.bundle
    const styles = image ? {
      header: {
        background: image && `url(${image})`,
        backgroundSize: image && '100%'
      }
    } : {}
 
    return {
      styles
    }
  },
  computed: {
  }
}
</script>
