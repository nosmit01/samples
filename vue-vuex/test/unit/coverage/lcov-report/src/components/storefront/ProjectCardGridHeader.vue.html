<template>
  <div class="container storefront-sort">
    <div class="row">
      <div class="storefront-sort-print visible-print-block">
        <div class="d-flex align-items-end">
          <img :src="flame" height="60" width="45" class="mr-1">
          <img :src="logo" height="45" width="171">
        </div>
      </div>
      <div class="col-md-7 d-flex align-items-center justify-content-between mb-2 mb-md-0 order-md-0 order-1 hidden-print">
        <h6 class="h5-md text-center text-md-left font-weight-normal w-100 m-0">
          <slot></slot>
        </h6>
      </div>
      <div
        class="col-md-5 d-flex align-items-center justify-content-md-end justify-content-center order-md-0 order-3 hidden-print"
      >
        <div class="d-flex align-items-center justify-content-md-end w-75" v-if="sortOptions.length > 1 && showSortOptions && !$route.query.q">
          <span class="text-right">Sort by</span>
          <select
            id="sortByDropdown"
            class="form-control ml-1 mr-1 ml-md-2 mr-md-0 w-75"
            v-model="selectedSortOption"
            @change="$emit('sort-option-change', selectedSortOption)"
          >
            <option :value="option.value" :key="option.value" v-for="option in sortOptions">{{ option.label }}</option>
          </select>
        </div>
        <AppToggleListView />
      </div>
 
      <div class="col-md-12 tabs-simple storefront-tabs order-md-0 order-2 hidden-print" v-if="selectedSubcategories && selectedSubcategories.length">
        <ul class="nav nav-pills nav-fill">
          <li
            class="nav-item"
            :class="{'active': selectedCategoryRoot === selectedCategory}"
          >
            <router-link
              :to="{ name: 'storefront:projects',
                    query: {
                      category: selectedCategoryRoot.id,
                      order: $route.query.order,
                      ns: 1,
                      view: getListQueryString
                    }
                  }"
              :class="{'active': selectedCategoryRoot === selectedCategory}"
              @click.native="browseCategory(selectedCategoryRoot.id, selectedCategoryRoot.name)"
              class="nav-link">All</router-link>
          </li>
          <li
            class="nav-item"
            v-for="subcategory in selectedSubcategories"
            :key="subcategory.id"
            :value="subcategory.id"
            :class="{'active': subcategory === selectedCategory}"
          >
            <router-link
              :to="{ name: 'storefront:projects',
                    query: {
                      category: subcategory.id,
                      order: $route.query.order,
                      ns: 1,
                      view: getListQueryString
                    }
                  }"
              :class="{'active': subcategory === selectedCategory}"
              @click.native="browseSubcategory(subcategory.id, subcategory.name)"
              class="nav-link">{{ subcategory.name}}</router-link>
          </li>
        </ul>
      </div>
    </div>
  </div>
</template>
 
<script>
import { createNamespacedHelpers } from 'vuex'
import AppToggleListView from '@/components/AppToggleListView'
import logo from '__STATIC__/images/responsive/logo/logo-black.svg'
import flame from '__STATIC__/images/responsive/logo/logo-flame-red.svg'
 
const { mapGetters } = createNamespacedHelpers('storefront')
 
export default {
  props: {
    categories: {
      type: Array,
      required: false
    },
    selectedCategory: {
      type: Object,
      required: false
    },
    selectedCategoryRoot: {
      type: Object,
      required: false
    },
    showSortOptions: {
      type: Boolean,
      default () {
        return true
      }
    },
    defaultSortOption: {
      type: String,
      required: false
    },
    sortOptions: {
      type: Array,
      default () {
        return []
      }
    }
  },
  components: {
    AppToggleListView
  },
  data () {
    return {
      logo,
      flame,
      selectedSortOption: this.$route.query.order || this.defaultSortOption
    }
  },
  computed: {
    selectedSubcategories () {
      return this.selectedCategoryRoot && this.getSubcategories(this.selectedCategoryRoot.id)
    },
    ...mapGetters(['getSubcategories', 'isListView']),
    getListQueryString () {
      return this.isListView ? 'list' : undefined
    }
  },
  methods: {
    browseCategory (categoryId, name) {
      this.$tracker.event('Click', `SF Menu Nav Type: ${name} ${categoryId}`, 'Storefront Menu')
    },
    browseSubcategory (categoryId, name) {
      this.$tracker.event('Click', `SF Menu Nav Subcategory`, 'Storefront Menu', {
        categoryId, name
      })
    },
    handleRouteChange () {
      this.selectedSortOption = this.$route.query.order || this.defaultSortOption
      this.scrollTabs()
    },
    scrollTabs () {
      this.$nextTick(() => {
        let width = this.$jq(window).width()
        let tabs = this.$jq('.tabs-simple a.active')
        if (tabs.length) {
          let scrollTo = tabs.position().left - (width / 3)
          this.$jq('.tabs-simple ul').scrollLeft(scrollTo)
        }
      })
    }
  },
  watch: {
    $route: {
      handler: 'handleRouteChange',
      immediate: true
    },
    selectedCategory: {
      handler: 'scrollTabs',
      immediate: true
    }
  }
}
</script>
