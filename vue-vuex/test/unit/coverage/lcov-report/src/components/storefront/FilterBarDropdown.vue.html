<template>
  <div
    class="caf-filter-bar-col-item caf-dropdown border-left border-right pl-0 pl-2 pr-1"
    @click="toggleDropdown"
    :class="{'open-dropdown' : dropdown.open, 'active' : active}"
  >
    <span class="hidden-md-down">{{label}}</span>
    <span class="hidden-md-down pl-1">
      <font-awesome-icon :icon="dropdown.open ? faAngleUp : faAngleDown" size="lg" />
    </span>
    <slot></slot>
  </div>
</template>
 
<script>
import FontAwesomeIcon from '@fortawesome/vue-fontawesome'
import faAngleDown from '@fortawesome/fontawesome-free-solid/faAngleDown'
import faAngleUp from '@fortawesome/fontawesome-free-solid/faAngleUp'
 
export default {
  props: {
    label: {
      type: String,
      required: true
    },
    active: {
      type: Boolean,
      required: false
    },
    open: {
      type: Boolean,
      required: false
    },
    isMobile: {
      type: Boolean,
      required: false
    },
    isMenu: {
      type: Boolean,
      default: false
    },
    isMenuOpen: {
      type: Boolean,
      default: false
    }
  },
  mounted () {
    window.addEventListener('click', this.clickedOutside, false)
  },
  beforeDestroy () {
    window.removeEventListener('click', this.clickedOutside, false)
  },
  components: {
    FontAwesomeIcon
  },
  data () {
    return {
      faAngleDown,
      faAngleUp,
      dropdown: { open: this.open || false }
    }
  },
  watch: {
    open (newVal) {
      this.dropdown.open = newVal
    },
    isMenuOpen (newVal) {
      this.dropdown.open = newVal
    }
  },
  methods: {
    clickedOutside () {
      if (!this.isMobile && !this.isMenuOpen) {
        const dropdownOpen = this.$el.classList.contains('open-dropdown')
        if (dropdownOpen) {
          this.dropdown.open = false
        }
      }
    },
    toggleDropdown () {
      if (!this.isMobile) {
        this.dropdown.open = !this.dropdown.open
      }
      this.dropdown.open && this.$emit('dropdown-open')
      this.$emit('menu-open', this.dropdown.open)
    }
  }
}
</script>
