<template>
  <div>
    <nav-header />
 
    <filter-bar />
 
    <project-card-grid-header
      class="pb-3 pt-4 pt-md-5"
      v-if="pageObj"
      :sortOptions="sortOptions"
      :showSortOptions="totalNewProjects > 1"
      @sort-option-change="reorder"
      :defaultSortOption="defaultSortOption"
    >
      <div v-if="loading">
        Loading projects...
      </div>
      <div v-else>
        Showing <strong>{{projects.length}}</strong> of <strong>{{ pageObj.count }} New Project{{ pageObj.count|pluralize }}</strong>
      </div>
    </project-card-grid-header>
 
    <app-grid-list
      :cards="projects"
      :loading="loading"
      :page="pageObj"
      @next-page="nextPage"
      @list-view="listView"
    >
      <template slot="no-results-text">
        There are no new projects right now.
      </template>
 
      <template slot="grid">
        <div class="col-md-4" v-for="project in projects" :key="project.id">
          <project-card
            :project="project"
            :selectedCategory="selectedCategory"
            :tracking="{label: 'SF Menu Project Cards', category: 'Storefront Menu'}"
            class="mb-3"
          ></project-card>
        </div>
      </template>
 
      <template slot="list-header">
        <div class="width-1 px-1">Save</div>
        <div class="width-6 px-1">Project</div>
        <div class="width-2 px-1">Estimated Time</div>
        <div class="width-3 px-1">Category</div>
      </template>
 
      <template slot="list">
        <project-row
          v-for="project in projects" :key="project.id"
          :project="project"
          :selectedCategory="selectedCategory"
          :tracking="{label: 'SF Menu Project List Row', category: 'Storefront Menu'}"
        ></project-row>
      </template>
 
    </app-grid-list>
 
  </div>
</template>
 
<script>
import { createNamespacedHelpers } from 'vuex'
 
import NavHeader from '@/components/storefront/NavHeader'
import FilterBar from '@/components/storefront/FilterBar'
import AppGridList from '@/components/AppGridList'
import ProjectCard from '@/components/storefront/ProjectCard'
import ProjectRow from '@/components/storefront/ProjectRow'
import ProjectCardGridHeader from '@/components/storefront/ProjectCardGridHeader'
import { pluralize } from '@/filters'
 
const { mapGetters, mapActions } = createNamespacedHelpers('storefront')
const { mapGetters: mapBreakpointGetters } = createNamespacedHelpers('breakpoints')
 
export default {
  components: {
    NavHeader,
    FilterBar,
    AppGridList,
    ProjectCardGridHeader,
    ProjectCard,
    ProjectRow
  },
  metaInfo: {
    title: 'Projects'
  },
  filters: {
    pluralize
  },
  data () {
    return {
      sortOptions: [
        {
          value: 'name',
          label: 'Name (A-Z)'
        },
        {
          value: 'popular',
          label: 'Most Popular'
        },
        {
          value: 'recent',
          label: 'Most Recent'
        }
      ],
      defaultSortOption: this.$route.query.order || 'recent',
      isDrawerOpen: false,
      filterByCategory: null
    }
  },
  computed: {
    selectedCategory () {
      return this.$route.query.category && this.getCategory(this.$route.query.category)
    },
    ...mapGetters([
      'totalNumProjects',
      'totalNewProjects',
      'rootCategories',
      'pageObj',
      'projects',
      'loading',
      'getCategory',
      'skills',
      'wishlistEnabled',
      'isListView'
    ]),
    ...mapBreakpointGetters(['isDesktop', 'isMobile'])
  },
  beforeRouteEnter (to, from, next) {
    next(vm => {
      vm.setQueryString()
      next()
    })
  },
  methods: {
    setQueryString () {
      if (this.$route.query.view) {
        this.setListView(true)
      }
      this.$router.replace({
        query: Object.assign({}, this.$route.query, {
          view: this.isListView ? 'list' : undefined
        })
      })
    },
    reorder (order) {
      const query = {...this.$route.query, order, ns: 1}
      this.$router.push({
        name: this.$route.name,
        replace: true,
        query
      })
    },
    fetchNewProjects () {
      this.fetchProjects({...this.$route.query, page: 1, isNew: true})
    },
    nextPage () {
      this.$tracker.event('Click', 'SF Menu Show More Projects', 'Storefront Menu')
      this.fetchProjects({
        ...this.$route.query,
        page: this.pageObj.number + 1,
        isNew: true,
        pageSize: this.$route.query.view === 'list' ? 16 : undefined
      })
    },
    ...mapActions(['fetchProjects', 'setListView']),
    drawerOpen (isOpen) {
      this.isDrawerOpen = isOpen
      this.filterByCategory = this.isMobile && !this.$route.query.skill
    },
    listView () {
      this.setListView(true)
      this.$router.push({
        name: this.$route.name,
        replace: true,
        query: Object.assign({}, this.$route.query, {
          ns: 1,
          view: 'list'
        })
      })
    }
  },
  watch: {
    // route changes are fired from nav toolbar or in the URL. We just pick up the changes here and update the store.
    $route: {
      handler: 'fetchNewProjects',
      immediate: true
    }
  }
}
</script>
