<template>
  <div class="caf-filter-bar-multiselect mt-3 mb-4 px-2">
    <div class="mb-3 font-weight-bold text-caf-red">
      <label for="selectAll" class="select-all-btn m-0">Select All</label>
      <input class="invisible" id="selectAll" type="checkbox" v-model="selectAll">
    </div>
    <ul>
      <li
        v-for="skill in skills"
        :key="skill.id"
      >
        <input type="checkbox" :id="skill.id" v-model="selected" :value="skill.id" />
        <label :for="skill.id" class="font-weight-semibold label mb-2 mb-md-1">{{ skill.name }}</label>
      </li>
    </ul>
    <div class="mb-3 font-weight-bold text-caf-red hidden-lg-up">
      <label for="selectAll2" class="select-all-btn">Select All</label>
      <input class="invisible" id="selectAll2" type="checkbox" v-model="selectAll">
    </div>
    <div class="d-flex justify-content-md-end justify-content-between pb-4">
      <button class="btn btn-secondary btn-sm mr-md-1 flex-md-first flex-last" @click="cancel">Cancel</button>
      <button
        class="btn btn-primary btn-sm mr-1 mr-md-0 width-3"
        @click="applySelection">
        Apply
      </button>
    </div>
  </div>
</template>
 
<script>
import { createNamespacedHelpers } from 'vuex'
 
import filter from 'lodash/filter'
import includes from 'lodash/includes'
import map from 'lodash/map'
import each from 'lodash/each'
import isArray from 'lodash/isArray'
import find from 'lodash/find'
import get from 'lodash/get'
 
const { mapGetters } = createNamespacedHelpers('storefront')
 
export default {
  props: {
    skills: {
      type: Array,
      required: true
    }
  },
  data () {
    return {
      selected: []
    }
  },
  mounted () {
    this.setSelected()
  },
  computed: {
    ...mapGetters(['isListView']),
    selectAll: {
      get: function () {
        return this.skills ? this.selected.length === this.skills.length : false
      },
      set: function (value) {
        this.selected = []
 
        if (value) {
          each(this.skills, skill => {
            this.selected.push(skill.id)
          })
        }
      }
    }
  },
  methods: {
    applySelection () {
      let query = ''
      let skillsNames = []
      each(this.selected, skillId => {
        query += `&skill=${skillId}`
        skillsNames.push(get(find(this.skills, {id: skillId}), 'name', ''))
      })
 
      if (this.isListView) {
        query += '&view=list'
      }
 
      this.$router.push({
        path: `/?order=name${query}&ns=1`,
        replace: true
      })
 
      this.$emit('menu-open', false)
      this.$tracker.event('Click', 'SF Menu Nav Skills: Apply Filter', 'Storefront Menu', {
        skills: skillsNames.join(', ')
      })
    },
    cancel () {
      this.selected = []
      this.$emit('menu-open', false)
    },
    setSelected () {
      let selectedSkills = this.$route.query.skill
      if (!selectedSkills) {
        return false
      }
 
      this.selected = map(
        filter(this.skills, skill =>
          isArray(selectedSkills)
            ? includes(map(selectedSkills, selectedSkill =>
              parseInt(selectedSkill)), skill.id)
            : parseInt(selectedSkills) === skill.id
        ), 'id')
    }
  }
}
</script>
