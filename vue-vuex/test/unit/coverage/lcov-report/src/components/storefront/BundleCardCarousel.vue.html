<template>
  <app-carousel :items="bundles" :alignToGrid="true" @navigate="trackNavigate" @swipe="trackSwipe">
    <template slot-scope="itemProps">
      <bundle-card
        class="mb-3"
        :bundle="itemProps.item"
        :show-description="false"
        :tracking="cardTracking"
        :isCarousel="true"></bundle-card>
    </template>
  </app-carousel>
</template>
 
<script>
import debounce from 'lodash/debounce'
import AppCarousel from '@/components/AppCarousel'
import BundleCard from '@/components/storefront/BundleCard'
 
export default {
  props: {
    bundles: {
      type: Array,
      required: true
    },
    carouselTracking: {
      type: Object,
      required: false
    },
    cardTracking: {
      type: Object,
      required: false
    }
  },
  components: {
    AppCarousel,
    BundleCard
  },
  methods: {
    trackNavigate () {
      if (this.carouselTracking) {
        const { label, category } = this.carouselTracking
        this.$tracker.event('Click', label, category)
      }
    },
    trackSwipe: debounce(function () {
      this.$tracker.event('Click', 'Featured: Navigate Carousel (mobile)', 'Featured')
    }, 800)
  }
}
</script>
