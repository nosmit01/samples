<template>
  <div>
 
    <nav-header />
 
    <filter-bar
      :disable-skill-filter="true"
      :disable-category-filter="true"
    />
 
    <div class="container pt-5 pb-3">
      <div class="row" v-if="!loading && pageObj.count > 0">
        <div class="col-md-8 d-flex align-items-center mb-2 mb-md-0 order-md-0 order-1">
          <h6 class="h5-md text-center text-md-left font-weight-normal w-100">
            <strong>{{ pageObj.count }}</strong> Bundle{{ pageObj.count|pluralize }}
          </h6>
        </div>
      </div>
    </div>
 
    <app-grid-list
      :cards="bundles"
      type="bundles"
      :loading="loading"
      :page="pageObj"
      @next-page="nextPage">
 
      <template slot="no-results-text">
        Sorry, we are unable to show any bundles at the moment.
      </template>
 
      <template slot="grid">
        <div class="col-md-4" v-for="bundle in bundles" :key="bundle.id">
          <bundle-card
            :bundle="bundle"
            :tracking="{label: 'SF Menu Bundle Cards', category: 'Storefront Menu'}"
            class="mb-3"
          ></bundle-card>
        </div>
      </template>
 
    </app-grid-list>
 
  </div>
</template>
 
<script>
import { createNamespacedHelpers } from 'vuex'
 
import { pluralize } from '@/filters'
 
import NavHeader from '@/components/storefront/NavHeader'
import FilterBar from '@/components/storefront/FilterBar'
import AppGridList from '@/components/AppGridList'
import BundleCard from '@/components/storefront/BundleCard'
 
const { mapGetters, mapActions } = createNamespacedHelpers('storefront')
const { mapGetters: mapBreakpointGetters } = createNamespacedHelpers('breakpoints')
 
export default {
  components: {
    NavHeader,
    FilterBar,
    AppGridList,
    BundleCard
  },
  metaInfo: {
    title: 'Project Menu: Bundles'
  },
  filters: {
    pluralize
  },
  data () {
    return {
    }
  },
  computed: {
    ...mapGetters([
      'pageObj',
      'bundles',
      'loading'
    ]),
    ...mapBreakpointGetters(['isDesktop', 'isMobile'])
  },
  methods: {
    handleRouteChange () {
      this.fetch()
    },
    fetch (page = 1) {
      this.fetchBundles({
        ...this.$route.query,
        page
      })
    },
    nextPage () {
      this.fetch(this.pageObj.number + 1)
    },
    ...mapActions(['fetchBundles'])
  },
  watch: {
    // route changes are fired from nav toolbar or in the URL. We just pick up the changes here and update the store.
    $route: {
      handler: 'handleRouteChange',
      immediate: true
    }
  }
}
</script>
