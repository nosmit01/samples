<template>
  <component
    ref="link"
    :is="this.isCall ? 'a' : 'router-link'"
    :href="this.isCall ? this.project.listing_url : undefined"
    :to="this.isProject ? this.detailLink : ''"
    @click.native="trackClick"
    class="caf-list-project container-fluid"
    @mouseleave.native="stopTimer"
    @mouseover.native="trackHover"
  >
    <div class="container d-flex caf-list-project-container">
      <div
        class="caf-list-project-wishlist width-1 px-1 px-lg-0 d-flex align-items-center"
        :class="{'enabled': wishlistEnabled}"
        @click.stop.prevent="toggleWishlist"
      >
        <span v-tooltip="{content: wishlistEnabled ? '' : 'You must be logged in as an organization to save this project.'}">
          <font-awesome-icon v-if="wishlisted" :icon="faStarSolid" />
          <font-awesome-icon v-if="!wishlisted" :icon="faStarOutline" />
        </span>
      </div>
      <div class="caf-list-project-group d-flex flex-column flex-md-row align-items-md-center">
        <div class="width-6 px-1 px-lg-0">
          <h6 class="caf-list-project-title">{{ projectName }}</h6>
          <div class="hidden-sm-down caf-list-project-description">{{ project.description}}</div>
        </div>
        <div class="width-2 px-1 px-lg-0 caf-list-project-time d-flex align-items-center">{{ project.timeframe }}</div>
        <div class="width-3 px-1 px-lg-0 caf-list-project-category d-flex align-items-center">
          <span class="color-dot mr-1 hidden-print" :style="styles.dot"></span>
          <font-awesome-icon class="visible-print-block mr-1" :style="styles.printDot" :icon="faCircle" size="lg" />
          {{ category.name }}
        </div>
      </div>
    </div>
  </component>
</template>
 
<script>
import FontAwesomeIcon from '@fortawesome/vue-fontawesome'
import faStarSolid from '@fortawesome/fontawesome-free-solid/faStar'
import faStarOutline from '@fortawesome/fontawesome-free-regular/faStar'
import faCircle from '@fortawesome/fontawesome-free-solid/faCircle'
 
import { tooltip } from '@/directives'
import { formatInteger, truncate } from '@/filters'
 
export default {
  props: {
    project: {
      type: Object,
      required: true
    },
    bundleItem: {
      type: Object,
      required: false
    },
    selectedCategory: {
      type: Object,
      required: false
    },
    selectedCategoryRoot: {
      type: Object,
      required: false
    },
    tracking: {
      type: Object,
      required: false
    },
    isCarousel: {
      type: Boolean,
      default: false
    },
    size: {
      type: String, // lg or sm
      required: false
    }
  },
  components: {
    FontAwesomeIcon
  },
  filters: {
    formatInteger,
    truncate
  },
  directives: {
    tooltip
  },
  mounted () {
    if (this.tracking && this.project.type === 'call') {
      this.$tracker.link(
        this.$refs.link,
        'Click', this.tracking.label, this.tracking.category
      )
    }
  },
  methods: {
    toggleWishlist (e) {
      if (this.$store.getters['storefront/wishlistEnabled']) {
        if (this.tracking) {
          this.$tracker.event('Click', `${this.tracking.label}: Save`, this.tracking.category)
        }
        this.$store.dispatch('storefront/toggleWishlist', this.project.id)
      }
    },
    trackHover () {
      if (this.tracking && !this.timer && !this.trackHoverComplete) {
        this.timer = window.setTimeout(() => {
          // double check in case of race condition
          if (!this.trackHoverComplete) {
            this.$tracker.event('Hover', this.tracking.label, this.tracking.category)
          }
          this.trackHoverComplete = true
          this.stopTimer()
        }, 1500)
      }
    },
    stopTimer () {
      if (this.tracking) {
        this.timer && window.clearTimeout(this.timer)
        this.timer = null
      }
    },
    trackClick () {
      if (this.project.type === 'project' && this.tracking) {
        this.$tracker.event('Click', this.tracking.label, this.tracking.category)
      }
    }
  },
  data () {
    const category = this.selectedCategoryRoot || this.project.category
 
    const color = (category && category.color) || '#0e1630'
    let gradient = ''
 
    if (color) {
      for (let i = 1; i < color.length; i++) {
        const value = parseInt(color[i], 16)
        const newValue = Math.ceil(Math.min(15, (value + (value / 3))))
        gradient += newValue.toString(16)
      }
      gradient = '#' + gradient
    }
 
    const styles = {
      dot: {
        background: color && `linear-gradient(77.71deg, ${color} -0.07%, ${gradient} 99.72%)`
      },
      printDot: {
        color: color && color
      }
    }
 
    let projectName = this.project.name
    if (this.project.type === 'call') {
      projectName += ' Call'
    }
 
    return {
      faStarSolid,
      faStarOutline,
      faCircle,
      styles,
      category,
      projectName
    }
  },
  computed: {
    isProject () {
      return this.project.type === 'project'
    },
    isCall () {
      return this.project.type === 'call'
    },
    detailLink () {
      const params = {
        id: this.project.id,
        slug: this.project.slug
      }
 
      const query = {
        category: this.selectedCategory && this.selectedCategory.id
      }
 
      if (this.bundleItem) {
        query.bundle = this.bundleItem.id
      }
 
      return {name: 'storefront:project', params, query}
    },
    wishlisted () {
      return this.$store.getters['storefront/isWishlisted'](this.project.id)
    },
    wishlistEnabled () {
      return this.$store.getters['storefront/wishlistEnabled']
    }
  }
}
</script>
