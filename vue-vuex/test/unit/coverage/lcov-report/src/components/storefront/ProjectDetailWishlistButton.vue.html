<template>
  <button
    type="button"
    class="btn w-100"
    :class="{'btn-selected': wishlisted, 'btn-secondary': !wishlisted, 'p-1 p-md-2': size === 'normal', 'btn-sm px-0': size === 'small'}"
    @click="toggle"
    :title="title"
    :disabled="!wishlistEnabled"
  >
    <font-awesome-icon :icon="faStar" />
    <span class="ml-md-1">{{ wishlisted ? 'Saved' : 'Save For Later' }}</span>
  </button>
</template>
 
<script>
import { createNamespacedHelpers } from 'vuex'
import FontAwesomeIcon from '@fortawesome/vue-fontawesome'
import faStar from '@fortawesome/fontawesome-free-regular/faStar'
 
const { mapGetters, mapActions } = createNamespacedHelpers('storefront')
 
export default {
  props: {
    project: {
      type: Object,
      required: true
    },
    size: {
      type: String,
      default: 'normal'
    }
  },
  components: {
    FontAwesomeIcon
  },
  data () {
    return {
      faStar
    }
  },
  computed: {
    wishlisted () {
      return this.project && this.isWishlisted(this.project.id)
    },
    title () {
      if (this.wishlistEnabled) {
        return this.wishlisted ? 'Remove from favorites' : 'Add to favorites'
      }
      return 'Sign in to save this project'
    },
    ...mapGetters({
      isWishlisted: 'isWishlisted',
      wishlistEnabled: 'wishlistEnabled'
    })
  },
  methods: {
    toggle () {
      this.$tracker.event('Click', 'Proj Details Save Project', 'Project Details')
      this.toggleWishlist(this.project.id)
    },
    ...mapActions(['toggleWishlist'])
  }
}
</script>
