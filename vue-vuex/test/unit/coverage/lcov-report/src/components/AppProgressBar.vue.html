<template>
  <ul class="progressbar-nodes">
    <li v-for="step in steps" :key="step.step" :class="stepClass(step.step)">
      <span class="step-icon"><font-awesome-icon :icon="faCheck" /></span>
      <span class="step-content">{{ step.label }}</span>
    </li>
  </ul>
</template>
 
<script>
import FontAwesomeIcon from '@fortawesome/vue-fontawesome'
import faCheck from '@fortawesome/fontawesome-free-solid/faCheck'
import map from 'lodash/map'
 
export default {
  props: {
    steps: {
      type: Array,
      required: true
    },
    active: {
      type: String,
      required: true
    }
  },
  components: {
    FontAwesomeIcon
  },
  methods: {
    stepClass (step) {
      const steps = map(this.steps, step => step.step)
      return {
        active: step === this.active,
        completed: steps.indexOf(this.active) > steps.indexOf(step)
      }
    }
  },
  data () {
    return { faCheck }
  }
}
</script>
