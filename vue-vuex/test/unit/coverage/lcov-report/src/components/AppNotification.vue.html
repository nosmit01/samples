<template>
  <div class="alert" :class="alertClass" role="alert" v-show="visible">
    <div class="container d-flex align-items-center">
      <span :class="iconClass" aria-hidden="true"></span>
      <p class="alert-content" v-html="message" ref="message"></p>
      <button
        v-if="dismissable"
        type="button"
        class="close"
        aria-label="Close"
        @click="handleClose"
        >
        <span class="ico-close"></span>
      </button>
    </div>
  </div>
</template>
 
<script>
export default {
  props: {
    message: {
      type: String,
      required: true
    },
    dismissable: {
      type: Boolean,
      default: true
    },
    level: {
      type: String,
      required: true,
      validator (value) {
        return ['info', 'success', 'warning', 'error'].indexOf(value) !== -1
      }
    },
    ignoreClose: { // used when you want to handle the hide/show toggle outside of component
      type: Boolean,
      default: false
    }
  },
  data () {
    let alertClass = {
      info: 'alert-info',
      success: 'alert-success',
      warning: 'alert-warning',
      error: 'alert-danger'
    }[this.level]
 
    if (this.dismissable) {
      alertClass += ' alert-dismissable'
    }
 
    return {
      visible: true,
      alertClass,
      iconClass: {
        info: 'ico-star',
        success: 'ico-success',
        warning: 'ico-warning-black',
        error: 'ico-warning-white'
      }[this.level]
    }
  },
  mounted () {
    const message = this.$refs.message
    // Pick up any tracking info on links embedded in the message
    this.$jq(message).find('a').on('click', (event) => {
      let elm = this.$jq(event.target)
      if (elm.data('tracking-label') && elm.data('tracking-category')) {
        this.$tracker.event('Click', elm.data('tracking-label'), elm.data('tracking-category'))
      }
    })
  },
  methods: {
    handleClose () {
      if (!this.ignoreClose) {
        this.visible = false
      }
      this.$emit('close')
    }
  }
}
</script>
