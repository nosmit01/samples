import { createNamespacedHelpers } from 'vuex'

import FontAwesomeIcon from '@fortawesome/vue-fontawesome'
 
import faChevronDown from '@fortawesome/fontawesome-free-solid/faChevronDown'
import faChevronUp from '@fortawesome/fontawesome-free-solid/faChevronUp'
 
import SidebarCard from '@/components/listing/SidebarCard'
 
import workflowStepMixin from '@/mixins/listing/workflowStepMixin'
 
import icoTip from '__STATIC__/images/responsive/icons/ico_tip_yellow.svg'
import icoCamera from '__STATIC__/images/responsive/icons/ico_camera_black.svg'
import imgListingCardExample from '__STATIC__/images/responsive/projects/listing-card-example.png'
import sampleCoverImage from '__STATIC__/images/responsive/projects/sample-cover-image.jpg'
import sampleProfileImage from '__STATIC__/images/responsive/projects/sample-profile-image.jpg'
 
const { mapGetters, mapActions } = createNamespacedHelpers('listing')
 
export default {
  mixins: [workflowStepMixin],
  components: {
    FontAwesomeIcon,
    SidebarCard,
    AppImageUpload,
    MobileInfoDialog,
    WorkflowStep
  },
  data () {
    return {
      icoTip,
      icoCamera,
      faChevronDown,
      faChevronUp,
      imgListingCardExample,
      sampleCoverImage,
      sampleProfileImage,
      showExample: false,
      coverImageInfo: `
      Your cover image appears on your Organization's profile page and
      on all of the projects and calls you post.`
    }
  },
  methods: {
    async save (onSuccess) {
      if (!this.isChanged) {
        return onSuccess()
      }
      const success = await this.saveCoverImage()
E      if (success) {
        onSuccess()
      }
    },
    handleBack () {
      this.save(() => {
        this.$router.push({
          name: 'listing:impact',
          params: this.$route.params
        })
      })
    },
    handleNext () {
      this.save(() => {
        this.$router.push({
          name: 'listing:final',
          params: this.$route.params
        })
      })
    },
    async handleExit () {
      this.save(() => {
        this.trackEvent('Submit', 'Save & Exit')
        this.redirectToDashboard()
      })
    },
    saveImageData (url, data) {
      this.images.cover_image = url
      // hack as image upload component controlled by jquery
      this.$jq('#id_cover_image').filestyle('buttonText', this.uploadLabel)
      this.multipartData.cover_image = data
      this.trackStepEvent('Click', 'Change Cover Image')
    },
    ...mapActions(['saveCoverImage'])
  },
  computed: {
    valid () {
      return !!this.images.cover_image || !!this.nonprofit.cover_image
    },
    uploadLabel () {
      return this.images.cover_image ? 'Change Image' : 'Choose an Image'
    },
    ...mapGetters([
      'multipartData',
      'images',
      'project',
      'projectType',
      'nonprofit'
    ])
  }
}
</script>
