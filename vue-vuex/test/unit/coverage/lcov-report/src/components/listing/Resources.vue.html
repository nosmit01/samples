<script>
import { createNamespacedHelpers } from 'vuex'
 
import FontAwesomeIcon from '@fortawesome/vue-fontawesome'

import FormTextarea from '@/components/forms/FormTextarea'
import SidebarCard from '@/components/listing/SidebarCard'
import WorkflowStep from '@/components/listing/WorkflowStep'
import MobileInfoDialog from '@/components/listing/MobileInfoDialog'
import MaxCounter from '@/components/listing/MaxCounter'
 
import faChevronDown from '@fortawesome/fontawesome-free-solid/faChevronDown'
import faChevronUp from '@fortawesome/fontawesome-free-solid/faChevronUp'
 
import icoTip from '__STATIC__/images/responsive/icons/ico_tip_yellow.svg'
 
import workflowStepMixin from '@/mixins/listing/workflowStepMixin'
 
import { markdown } from '@/filters'
 
const { mapGetters, mapActions } = createNamespacedHelpers('listing')
 
export default {
  mixins: [workflowStepMixin],
  components: {
    FontAwesomeIcon,
    FormInput,
    FormTextarea,
    FormGroup,
    MaxCounter,
    SidebarCard,
    WorkflowStep,
    MobileInfoDialog
  },
  data () {
    return {
      faChevronUp,
      faChevronDown,
      icoTip,
      isTemplate: true,
      additionalResourcesFocused: false,
      existingResourcesFocused: false,
      maxResources: 500,
      maxExistingResources: 60,
      maxAdditionalResources: 60,
      sidebarCardStyle: {},
      resourcesInfo: `
      Identifying the resources you will provide shows volunteers that you are committed
      to the success of this project.`,
      resourcesExample: `
      We currently have a report with our metrics from last year, which should make it
      easy for you to get started. We also have raw data for this year, and the ability
      to provide any other information you need.`
    }
  },
  mounted () {
    this.isTemplate = !this.formData.resources
    this.sidebarCardStyle = {
      position: 'absolute',
      top: `${this.$refs.resources.offsetTop}px`
    }
  },
  methods: {
    markdown,
    confirmChangeEditMode () {
      return this.$confirm.show({
        title: 'Are you sure you want to change your format?',
        message: 'Any text that you entered will be lost',
        buttons: {
          confirm: 'Yes, Proceed',
          cancel: 'No, Cancel'
        }
      })
    },
    async changeEditMode (isTemplate) {
      // at this point launch modal if isChanged
      if (isTemplate) {
        if (this.formData.resources) {
          const confirmed = await this.confirmChangeEditMode()
          if (!confirmed) {
            return
          }
        }
        this.formData.resources = ''
      } else {
        if (this.formData.additional_resources || this.formData.existing_resources) {
          const confirmed = await this.confirmChangeEditMode()
          if (!confirmed) {
            return
          }
        }
        this.formData.additional_resources = ''
        this.formData.existing_resources = ''
      }
      this.isTemplate = isTemplate
    },
    async save (onSuccess, required) {
      if (!this.isChanged) {
        return onSuccess()
      }
      const success = await this.saveProjectResources({ required })
E      if (success) {
        onSuccess()
      }
    },
    handleBack () {
      this.save(() => {
        this.$router.push({
          name: 'listing:deliverables',
          params: this.$route.params
        })
      })
    },
    handleNext () {
      this.save(() => {
        this.$router.push({
          name: 'listing:impact',
          params: this.$route.params
        })
      }, true)
    },
    handleExit () {
      this.save(() => {
        this.trackExit()
        this.redirectToDashboard()
      })
    },
    ...mapActions(['saveProjectResources'])
  },
  computed: {
    valid () {
      return !!this.formData.resources || (!!this.formData.additional_resources && !!this.formData.existing_resources)
    },
    additionalResourcesMode () {
      return this.formData.additional_resources.length > 0 && !this.additionalResourcesFocused ? 'text' : 'input'
    },
    existingResourcesMode () {
      return this.formData.existing_resources.length > 0 && !this.existingResourcesFocused ? 'text' : 'input'
    },
    ...mapGetters([
      'formData',
      'submitting',
      'project',
      'projectType'
    ])
  }
}
</script>
