<template>
  <div
    class="modal fade"
    tabindex="-1"
    role="dialog"
    aria-labelledby="dialog"
    aria-hidden="true"
    >
    <div class="modal-dialog" role="document">
      <div class="modal-content">
        <button type="button" class="close" data-dismiss="modal" aria-label="Close">
          <span class="ico-close-gray" aria-hidden="true"></span>
        </button>
        <preview-overlay></preview-overlay>
      </div>
    </div>
  </div>
</template>
 
<script>
import PreviewOverlay from '@/components/listing/PreviewOverlay'
export default {
  components: {
    PreviewOverlay
  },
  data () {
    return {
      $modal: null
    }
  },
  mounted () {
    this.$modal = this.$jq(this.$el)
    this.$modal.on('hidden.bs.modal', () => {
      this.$emit('close')
      return true
    })
  },
  methods: {
    show () {
      this.$modal.modal('show')
      this.$emit('show')
    }
  }
}
</script>
