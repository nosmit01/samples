<script>
import { createNamespacedHelpers } from 'vuex'
 
import FontAwesomeIcon from '@fortawesome/vue-fontawesome'
 
import faChevronDown from '@fortawesome/fontawesome-free-solid/faChevronDown'
import faChevronUp from '@fortawesome/fontawesome-free-solid/faChevronUp'
 
import { formatInteger } from '@/filters'
import { tooltip } from '@/directives'
 
import FormGroup from '@/components/forms/FormGroup'
import FormTextarea from '@/components/forms/FormTextarea'
import SidebarCard from '@/components/listing/SidebarCard'
import MaxCounter from '@/components/listing/MaxCounter'
import WorkflowStep from '@/components/listing/WorkflowStep'
import MobileInfoDialog from '@/components/listing/MobileInfoDialog'
 
import icoTip from '__STATIC__/images/responsive/icons/ico_tip_yellow.svg'
 
import workflowStepMixin from '@/mixins/listing/workflowStepMixin'
 
const { mapGetters, mapActions } = createNamespacedHelpers('listing')
 
export default {
  mixins: [workflowStepMixin],
  components: {
    FontAwesomeIcon,
    FormGroup,
    FormTextarea,
    MaxCounter,
    SidebarCard,
    WorkflowStep,
    MobileInfoDialog
  },
  directives: {
    tooltip
  },
  data () {
    return {
      icoTip,
      faChevronDown,
      faChevronUp,
      maxGoals: 500,
      maxImpactStatement: 200,
      goalsCardStyle: {
        position: 'absolute',
        top: '85px'
      },
      goalsInfo: `
      Explaining how this project will support your mission helps volunteers understand their
      individual potential to make a difference.`,
      goalsExample: `
      Sharing specific stories about how donations translate into tangible benefits
      for the community we serve is one of the most effective ways of showing our
      supporters how much we appreciate them, and of encouraging them to keep
      giving.`,
      impactStatementStyle: {},
      impactStatementCardStyle: {
        position: 'absolute',
        top: '330px'
      },
      impactStatementInfo: `
      Describing what else you can do with your money you will save on this project
      helps volunteers understand the immediate value they can offer to your
      Organization.`,
      impactStatementTooltip: `
        This is the value we've estimated for this project.
        <a href="https://medium.com/@catchafirenyc/evolving-how-we-calculate-catchafires-project-values-8eaa200b906"
        rel="noopener" target="_blank">Learn more</a> about how we calculate project savings.`
    }
  },
  mounted () {
    const labelWidth = Math.round(
      parseFloat(window.getComputedStyle(this.$refs.impactStatementLabel).width)
    ) + 4
    let height = `${this.$refs.impactStatementInput.scrollHeight}px`
    this.impactStatementStyle = {
      textIndent: `${labelWidth}px`,
      // overflow: 'visible',
      height
    }
    // console.log('IS HEIGHT 1', height)
    // ensure scrollheight has correctly updated so we can set the correct height
    // of the textarea
    this.$nextTick(() => {
      height = `${this.$refs.impactStatementInput.scrollHeight}px`
      this.impactStatementStyle.height = height
      // console.log('IS HEIGHT 2', height)
    })
  },
  methods: {
    async save (onSuccess, required) {
      if (!this.isChanged) {
        return onSuccess()
      }
      const success = await this.saveProjectImpact({ required })
E      if (success) {
        onSuccess()
      }
    },
    toggleGoalsExample (show) {
      if (show) {
        this.trackShowExample()
        this.$refs.impactStatementCard.close()
        this.goalsCardStyle.top = '30px'
        this.impactStatementCardStyle.top = '430px'
      } else {
        this.goalsCardStyle.top = '85px'
        this.impactStatementCardStyle.top = '330px'
      }
    },
    toggleImpactStatementExample (show) {
      if (show) {
        this.trackShowExample()
        this.$refs.goalsCard.close()
      }
    },
    handleBack () {
      this.save(() => {
        this.$router.push({
          name: 'listing:resources',
          params: this.$route.params
        })
      })
    },
    handleNext () {
      this.save(() => {
        this.$router.push({
          name: 'listing:image',
          params: this.$route.params
        })
      }, true)
    },
    handleExit () {
      this.save(() => {
        this.trackEvent('Submit', 'Save & Exit')
        this.redirectToDashboard()
      })
    },
    formatInteger,
    ...mapActions(['saveProjectImpact'])
  },
  computed: {
    valid () {
      return !!this.formData.goals && !!this.formData.impact_statement
    },
    impactStatementExample () {
      return `This project will save us $${this.savings},
      allowing us to purchase school supplies for more children in need.`
    },
    goalsLength () {
      return Math.min(this.maxGoals, this.formData.goals.length)
    },
    savings () {
      return this.projectType ? this.formatInteger(this.projectType.savings) : ''
    },
    ...mapGetters([
      'formData',
      'projectType',
      'project',
      'submitting'
    ])
  }
}
</script>
