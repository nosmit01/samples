<template>
  <div class="rounded p-3 pr-4 pb-4 bg-color-white my-1 ml-md-4 mr-md-1 text-color-emperor small">
    <img v-if="icon" :src="icon" class="mb-2">
    <p class="mb-0">
      <slot></slot>
    </p>
    <div class="font-weight-semibold mt-2" v-show="!showExample" v-if="hasExample">
      <a ref="showExample"  href="#" @click.prevent="toggleExample(true)" class="text-black">
        See an example
        <font-awesome-icon :icon="faChevronDown"></font-awesome-icon>
      </a>
    </div>
    <div class="font-weight-semibold mt-2" v-show="showExample" v-if="hasExample">
      <a ref="hideExample" href="#" @click.prevent="toggleExample(false)" class="text-black">
        Hide example
        <font-awesome-icon :icon="faChevronUp"></font-awesome-icon>
      </a>
    </div>
    <p class="font-italic mt-1" v-show="showExample" v-if="hasExample">
      <slot name="example"></slot>
    </p>
  </div>
</template>
 
<script>
import FontAwesomeIcon from '@fortawesome/vue-fontawesome'
 
import faChevronDown from '@fortawesome/fontawesome-free-solid/faChevronDown'
import faChevronUp from '@fortawesome/fontawesome-free-solid/faChevronUp'
 
export default {
  components: {
    FontAwesomeIcon
  },
  props: {
    icon: {
      required: false
    }
  },
  data () {
    return {
      faChevronDown,
      faChevronUp,
      showExample: false,
      hasExample: !!this.$slots.example
    }
  },
  methods: {
    close () {
      this.toggleExample(false)
    },
    toggleExample (show) {
      this.showExample = show
      this.$emit(show ? 'show-example' : 'hide-example')
    }
  }
}
</script>
