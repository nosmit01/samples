import FontAwesomeIcon from '@fortawesome/vue-fontawesome'

import faChevronUp from '@fortawesome/fontawesome-free-solid/faChevronUp'
 
import AppImageUpload from '@/components/AppImageUpload'
import FormInput from '@/components/forms/FormInput'
import FormGroup from '@/components/forms/FormGroup'
import FormTextarea from '@/components/forms/FormTextarea'

import SidebarCard from '@/components/listing/SidebarCard'
import WorkflowStep from '@/components/listing/WorkflowStep'
import MobileInfoDialog from '@/components/listing/MobileInfoDialog'
 
import logoPlaceholder from '__STATIC__/images/responsive/placeholders/placeholder_org_logo.png'

import workflowStepMixin from '@/mixins/listing/workflowStepMixin'
 
import icoTip from '__STATIC__/images/responsive/icons/ico_tip_yellow.svg'
 
const { mapGetters, mapActions } = createNamespacedHelpers('listing')
 
export default {
  mixins: [workflowStepMixin],
  components: {
    AppImageUpload,
    FontAwesomeIcon,
    FormGroup,
    FormInput,
    FormTextarea,
    SidebarCard,
    MaxCounter,
    MobileInfoDialog,
    WorkflowStep
  },
  data () {
    return {
      icoTip,
      faChevronDown,
      faChevronUp,
      logoPlaceholder,
      imagePlaceholder,
      maxAnecdote: 250,
      anecdoteCardStyle: {
        position: 'absolute',
        top: '100px'
      },
      profileCardStyle: {
        position: 'absolute',
        top: '390px'
      },
      anecdoteInfo: 'Providing an interesting tidbit about your Organization helps break the ice with potential volunteers.',
      anecdoteExample: `The name Catchafire was inspired by the 1973 album Catch a Fire by The Wailers,
      and we have a framed photo of the original album cover art in our office!`,
      profileInfo: `
      Completing your profile information improves how your project
      is displayed, and it helps volunteers see who they will be working
      with.`
    }
  },
  methods: {
    async save (onSuccess) {
      if (!this.isChanged) {
        return onSuccess()
      }
      const success = await this.saveFinalDetails()
E      if (success) {
        onSuccess()
      }
    },
    toggleAnecdoteExample (show) {
      this.profileCardStyle.top = show ? '410px' : '390px'
      this.trackShowExample()
    },
    handleBack () {
      this.save(() => {
        this.$router.push({
          name: 'listing:image',
          params: this.$route.params
        })
      })
    },
    handleNext () {
      this.save(() => {
        this.$router.push({
          name: 'listing:review',
          params: this.$route.params
        })
      })
    },
    handleExit () {
      this.save(() => {
        this.trackExit()
        this.redirectToDashboard()
      })
    },
    saveLogo (url, data) {
      this.images.logo = url
      // hack as image upload component controlled by jquery
      this.$jq('#id_logo').filestyle('buttonText', this.logoWidgetLabel)
      this.multipartData.logo = data
      this.trackStepEvent('Click', 'Change Logo')
    },
    saveImage (url, data) {
      this.images.image = url
      // hack as image upload component controlled by jquery
      this.$jq('#id_image').filestyle('buttonText', this.profileWidgetLabel)
      this.multipartData.image = data
      this.trackStepEvent('Click', 'Change Profile Image')
    },
    ...mapActions(['saveFinalDetails'])
  },
  computed: {
    logoWidgetTitle () {
      return this.images.logo ? 'Update Logo' : 'Add Logo'
    },
    logoWidgetLabel () {
      return this.images.logo ? 'Change Your Logo' : 'Choose An Image'
    },
    profileWidgetLabel () {
      return this.images.image ? 'Change Your Image' : 'Choose An Image'
    },
    ...mapGetters([
      'formData',
      'multipartData',
      'images',
      'submitting',
      'profile',
      'project',
      'nonprofit'
    ])
  }
