<template>
  <div class="caf-list-toggle pl-md-3">
    <div class="caf-list-toggle-grid" :class="{ active: !$route.query.view }" @click="toggleListView(false)"></div>
    <div class="caf-list-toggle-pipe"></div>
    <div class="caf-list-toggle-list" :class="{ active: $route.query.view }" @click="toggleListView(true)">
      <NewFeatureIndicator v-if="!isListView" storage-key="list-view-dot" expires="2018-11-25" position="top: -2px; right: -7px" />
    </div>
  </div>
</template>
<script>
import { createNamespacedHelpers } from 'vuex'
 
import NewFeatureIndicator from '@/components/NewFeatureIndicator'
 
const { mapGetters, mapActions } = createNamespacedHelpers('storefront')
 
export default {
  components: {
    NewFeatureIndicator
  },
  computed: {
    ...mapGetters(['isListView'])
  },
  methods: {
    ...mapActions(['setListView']),
    toggleListView (isListView) {
      this.setListView(isListView)
      this.$tracker.event('Click', `SF Menu ${isListView ? 'List' : 'Grid'}: Select`, 'Storefront Menu')
      this.$router.push({
        name: this.$route.name,
        replace: true,
        query: Object.assign({}, this.$route.query, {
          ns: 1,
          view: isListView ? 'list' : undefined
        })
      })
    }
  }
}
</script>
