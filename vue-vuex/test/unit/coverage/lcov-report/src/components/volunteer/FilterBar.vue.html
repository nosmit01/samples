  <div class="caf-filter-bar-container sticky-top">
    <div
      ref="filterBar"
      class="container-fluid caf-filter-bar bg-color-white"
      :class="{'fixed-top': mobileFilterActive && isMobile}"
      v-sticky-top="{elTop : 0, scrollTop : isMobile ? 289 : 276, bodyOffset : 50, isMobile, isOpen : mobileFilterActive}"
    >
      <div
        class="container h-100 d-flex align-items-center justify-content-between text-caf-blue font-weight-semibold"
      >
        <div class="caf-filter-bar-col"
             :class="{'hide' : expandSearch}">
          <div
            class="caf-filter-bar-col-item mobile-filter border-left border-right px-2 hidden-lg-up"
            :class="{'active' : mobileFilterActive}"
            @click="$tracker.event('Click', 'Live Listings Page: Filter (Mobile)', 'Find a Project Page'); menuOpen('mobile-filter')"
          >
            <span class="pr-1"><font-awesome-icon :icon="faFilter" size="lg" /></span> Filter
          </div>
          <filter-bar-filter
            v-if="!isMobile"
            v-for="(filter, i) in filters"
            :key="filter.id"
            :filter="filter"
            :class="{'border-left' : i === 0, active : filter.selected}"
            :showCount="filter.type !== 'length'"
            :styles="{minWidth: '180px'}"
            @menu-open="$tracker.event('Click', 'Live Listings Page: ' + filter.type.replace(/^\w/, c => c.toUpperCase()) + ' Menu', 'Find a Project Page'); menuOpen(filter.type)"
          />
          <div v-if="!isMobile && !expandSearch" class="font-size-14 text-caf-red d-flex align-items-center pl-2">
            <span class="caf-filter-bar-settings font-weight-bold font-main" @click="clearSettings">Clear</span>
            <span v-if="isVolunteer" class="font-weight-normal text-caf-blue px-1">|</span>
            <span v-if="isVolunteer" class="caf-filter-bar-settings font-weight-bold font-main" @click="saveSettings">Save settings</span>
          </div>
        </div>
 
        <div class="caf-filter-bar-col"
             :class="{'expand' : isMobile && expandSearch}"
             :style="{width: expandSearchWidth, right: 0}"
             v-click-outside="resetSearchForm"
        >
          <div class="caf-filter-bar-col-item search border-right w-100"
               :class="{'active' : expandSearch, 'border-left' : !expandSearch}"
          >
            <form @submit.prevent="search(null)" class="w-100">
              <input
                class="search-input vol-search ml-2 pr-2 text-caf-blue"
                type="text"
                placeholder="Search all"
                ref="searchTerm"
                v-model="searchTerm"
                @click="$refs.searchTerm.select()"
                @focus="setExpandSearch"
              />
            </form>
            <span
              v-if="!expandSearch"
              class="search-icon ml-1 mr-2"
              @click="setExpandSearch"
            >
              <font-awesome-icon :icon="faSearch" flip="horizontal" size="lg" />
            </span>
            <span
              v-if="expandSearch"
              class="search-icon ml-1 mr-2"
              @click="setCollapseSearch"
            >
              <h6 class="mb-0 mr-1 font-size-13 font-weight-semibold text-nowrap text-uppercase letter-spacing-1">Clear search</h6>
              <font-awesome-icon :icon="faTimes" size="lg" />
            </span>
          </div>
        </div>
      </div>
    </div>
 
    <app-notification
      v-if="isSettingsSaved"
      message="Your cause and skill selections have been saved to your profile. You can also update these in your <a href='/settings/projects/' data-tracking-label='Live Listings Page: Settings Link' data-tracking-category='Find a Project Page'>Settings</a>."
      :dismissable="true"
      level="success"
      :ignoreVisible="true"
      @close="resetIsSettingsSaved"
      v-sticky-top="{elTop : 50, scrollTop : isMobile ? 289 : 276, class : 'fixed-notification-top'}"
    />
 
    <!-- Menu container that holds:
        - filter menu options on desktop and mobile
        - filter dropdowns on mobile (filter dropdowns on desktop are in the code above)
        - filter clear/ save settings on mobile (clear/ save settings on desktop are in the code above)
        - save settings notification on mobile (save settings notification on desktop is in the code above)
     -->
    <filter-drop-menu
      :isOpen="!isUndefined(filterSelected) || mobileFilterActive"
      :isMobile="isMobile"
      :isSearchOpen="expandSearch"
      :type="get(filterSelected, 'type')"
      :bg="get(filterSelected, 'menuBg')"
      @menu-open="menuOpen"
    >
      <!-- Clear/ save settings for mobile -->
      <div v-if="isMobile && !expandSearch" class="font-size-14 caf-filter-bar-settings-container font-weight-bold text-caf-red d-flex align-items-center pl-2 py-2">
        <span class="caf-filter-bar-settings font-weight-bold font-main" @click="clearSettings">Clear</span>
        <span v-if="isVolunteer" class="font-weight-normal text-caf-blue px-1">|</span>
        <span v-if="isVolunteer" class="caf-filter-bar-settings font-weight-bold font-main" @click="saveSettings">Save settings</span>
      </div>
 
      <!-- Filter dropdowns for mobile -->
      <div v-if="isMobile" class="caf-filter-bar-menu-items d-flex align-items-center px-2">
        <div
          v-for="filter in filters"
          :key="filter.id"
          class="caf-filter-bar-menu-item mr-3"
          :class="{'active' : filter.selected, 'disabled' : expandSearch, 'pr-3': filter.type === 'length'}"
          @click="$tracker.event('Click', 'Live Listings Page: ' + filter.type.replace(/^\w/, c => c.toUpperCase()) + ' Menu', 'Find a Project Page'); menuOpen(filter.type)">
          <div class="d-flex justify-content-between">
            <span class="pr-1">
              <font-awesome-icon :icon="filter.icon" size="lg" />
            </span>
            <span>{{filter.label}}</span>
            <span
              v-if="filter.type !== 'length'"
              class="pl-1"
            >
              {{filter.count > 0 ? filter.count : ''}}
            </span>
          </div>
        </div>
      </div>
 
      <!-- filter menu options for desktop and mobile -->
      <filter-multi-select
        v-if="get(filterSelected, 'type') === filter.type"
        v-for="filter in filterMenu"
        :key="filter.id"
        :type="filter.type"
        :items="getOptions(filter.type)"
        :settings="settings[filter.type] || []"
        :isMobile="isMobile"
        @menu-open="menuOpen(filter.type)"
        @apply-selection="applySelection"
      />
 
      <!-- length filter menu options. This is better than adding a bunch of conditionals-->
      <filter-length-multi-select
        v-if="get(filterSelected, 'type') === 'length'"
        :items="getOptions('length')"
        :settings="settings.length || []"
        :isMobile="isMobile"
        @menu-open="menuOpen('length')"
        @apply-selection="applySelection"
      />
    </filter-drop-menu>
  </div>
</template>
 
<script>
import { createNamespacedHelpers } from 'vuex'
import FilterBarFilter from '@/components/volunteer/FilterBarFilter'
import FilterDropMenu from '@/components/volunteer/FilterDropMenu'
import FilterMultiSelect from '@/components/volunteer/FilterMultiSelect'
import FilterLengthMultiSelect from '@/components/volunteer/FilterLengthMultiSelect'
import FontAwesomeIcon from '@fortawesome/vue-fontawesome'
import AppNotification from '@/components/AppNotification'
import { faSearch } from '@fortawesome/pro-solid-svg-icons/faSearch'
import { faFilter } from '@fortawesome/pro-solid-svg-icons/faFilter'
import { faPencilRuler } from '@fortawesome/pro-regular-svg-icons/faPencilRuler'
import { faHeart } from '@fortawesome/pro-regular-svg-icons/faHeart'
import { faClock } from '@fortawesome/pro-regular-svg-icons/faClock'
import { faTimes } from '@fortawesome/pro-regular-svg-icons/faTimes'
import { faPencilRuler as faPencilRulerLight } from '@fortawesome/pro-light-svg-icons/faPencilRuler'
import { faHeart as faHeartLight } from '@fortawesome/pro-light-svg-icons/faHeart'
import filter from 'lodash/filter'
import map from 'lodash/map'
import assign from 'lodash/assign'
import find from 'lodash/find'
import get from 'lodash/get'
import isUndefined from 'lodash/isUndefined'
import each from 'lodash/each'
import { stickyTop, clickOutside } from '@/directives'
import { stickyTopMx } from '@/mixins'
 
const { mapGetters, mapActions } = createNamespacedHelpers('volunteer')
const { mapGetters: mapBreakpointGetters } = createNamespacedHelpers('breakpoints')
 
export default {
  components: {
    FilterBarFilter,
    FilterDropMenu,
    FilterMultiSelect,
    FilterLengthMultiSelect,
    FontAwesomeIcon,
    AppNotification
  },
  data () {
    return {
      find,
      get,
      filter,
      isUndefined,
      faSearch,
      faFilter,
      faTimes,
      searchTerm: '',
      order: 'name',
      expandSearch: null,
      mobileFilterActive: false,
      filters: [
        { type: 'causes', label: 'Causes', open: false, icon: faHeart, menuBg: faHeartLight, count: 0, useCount: true, selected: false },
        { type: 'skills', label: 'Skills', open: false, icon: faPencilRuler, menuBg: faPencilRulerLight, count: 0, useCount: true, selected: false },
        { type: 'length', label: 'Length', open: false, icon: faClock, selected: false }
      ]
    }
  },
  directives: {
    stickyTop,
    clickOutside
  },
  mixins: [stickyTopMx],
  mounted () {
    // anonymous (logged out) users
    if (!this.isAuth && !this.$route.query.hasOwnProperty('page')) {
      const query = {'page': 1}
      this.$router.push({
        name: 'volunteer:projects',
        replace: true,
        query
      })
    }
    // pass a boolean to ignore grabbing the settings
    // ie. deep-linked URL with querystrings
    if (this.$route.query.hasOwnProperty('init')) { return }
    !this.$route.query.hasOwnProperty('page') && this.fetchSettings()
  },
  computed: {
    hasSearched () {
      return !!this.$store.getters['volunteer/searchTerm']
    },
    filterSelected () {
      return find(this.filters, 'selected')
    },
    filterMenu () {
      return filter(this.filters, item => item.type !== 'length')
    },
    expandSearchWidth () {
      let width = 'auto'
 
      if (this.expandSearch) {
        width = '100%'
      } else {
        if (this.isMobile) {
          width = 'auto'
        } else {
          width = '200px'
        }
      }
 
      return width
    },
    ...mapGetters([
      'isAuth',
      'settings',
      'causes',
      'skills',
      'length',
      'isSettingsSaved',
      'projects',
      'isVolunteer'
    ]),
    ...mapBreakpointGetters(['isDesktop', 'isMobile'])
  },
  watch: {
    $route: {
      handler: 'updateSearchAndSettings',
      immediate: true
    },
    settings (n) {
      this.setSelectedCount()
 
      // update the URL based on returned settings
      const query = {
        'name_filter': '',
        'skill_filter': n['skills'],
        'cause_filter': n['causes'],
        'type_filter': n['length'],
        'page': 1
      }
      this.$router.push({
        name: 'volunteer:projects',
        replace: true,
        query
      })
    }
  },
  methods: {
    getOptions (type) {
      return this[type]
    },
    setSelectedCount () {
      this.filters = map(this.filters, filter => {
        return assign({}, filter, {
          count: get(this, `settings[${filter.type}]`, []).length
        })
      })
    },
    updateSearchAndSettings () {
      const { name_filter, cause_filter, skill_filter, type_filter } = this.$route.query || {}
      this.searchTerm = name_filter || ''
      this.expandSearch = !!this.searchTerm
      this.updateSearchTerm(this.searchTerm)
 
      let causes = cause_filter || []
      let skills = skill_filter || []
      let length = type_filter || [1, 2]
      this.settings.causes = (typeof causes === 'string') ? [causes] : causes
      this.settings.skills = (typeof skills === 'string') ? [skills] : skills
      this.settings.length = (typeof length === 'string') ? [length] : length
      this.updateSettings(this.settings)
      this.setSelectedCount()
    },
    search () {
      this.$tracker.event('Submit', 'Live Listings Page: Search', 'Find a Project Page')
      if (this.isMobile) {
        this.mobileFilterActive = false
        this.menuOpen(false)
      }
      this.updateSearchTerm(this.searchTerm)
 
      const query = {'name_filter': this.searchTerm, 'page': 1}
      this.$router.push({
        name: 'volunteer:projects',
        replace: true,
        query
      })
    },
    setExpandSearch () {
      this.$tracker.event('Click', 'Live Listings Page: Search', 'Find a Project Page')
      this.expandSearch = true
      this.$refs.searchTerm.focus()
      document.dispatchEvent(new Event('input:focus'))
    },
    setCollapseSearch () {
      this.$tracker.event('Click', 'Live Listings Page: Clear Search', 'Find a Project Page')
      this.expandSearch = false
      this.searchTerm = ''
      if (this.hasSearched) {
        this.updateSearchTerm(this.searchTerm)
        this.clearSettings(true)
        if (!this.isAuth) {
          const query = {}
          this.$router.push({
            name: 'volunteer:projects',
            replace: true,
            query
          })
          this.$emit('fetch-projects')
        }
        !!this.isAuth && this.fetchSettings()
      }
    },
    resetSearchForm () {
      if (!this.isMobile) {
        this.expandSearch = false
        this.searchTerm = ''
      }
    },
    menuOpen (type) {
      this.resetIsSettingsSaved() // hide the app notification if visible
      if (type === 'mobile-filter') { // user clicked `Filter` button on mobile
        this.mobileFilterActive = !this.mobileFilterActive
        // select first filter, and set others to false
        this.filters = map(this.filters, (filter, i) => {
          if (i === 0) {
            return assign({}, filter, {
              selected: this.mobileFilterActive
            })
          }
 
          return assign({}, filter, {
            selected: false
          })
        })
      } else { // user selected a filter
        this.mobileFilterActive = !!type
        // toggle selected filter and set others to false
        this.filters = map(this.filters, filter => {
          if (filter.type === type) {
            return assign({}, filter, {
              selected: this.isMobile ? true : !filter.selected
            })
          }
 
          return assign({}, filter, {
            selected: false
          })
        })
      }
 
      this.stickyTopMx({
        el: this.$refs.filterBar,
        elTop: 0,
        scrollTop: 276,
        stickyClass: 'fixed-top',
        isMobile: this.isMobile && (this.mobileFilterActive),
        mobileClass: 'fixed-top',
        bodyOffset: 50,
        hasBodyScroll: true,
        isOpen: this.mobileFilterActive
      })
    },
    applySelection (type, selections) {
      this.settings[type] = selections
      this.updateSettings(this.settings)
 
      this.filters = map(this.filters, filter => {
        if (filter.type === type) {
          return assign({}, filter, {
            count: selections.length
          })
        }
 
        return filter
      })
 
      const query = {
        name_filter: '',
        skill_filter: this.settings['skills'],
        cause_filter: this.settings['causes'],
        type_filter: this.settings['length'],
        page: 1
      }
      this.$router.push({
        name: 'volunteer:projects',
        replace: true,
        query
      })
    },
    saveSettings () {
      this.$tracker.event('Click', 'Live Listings Page: Save Settings', 'Find a Project Page')
      this.mobileFilterActive = false
      this.menuOpen(false)
      // called it storeSaveSettings to differentiate from saveSettings used in the template
      this.storeSaveSettings(this.settings)
    },
    clearSettings (ignoreFetch) {
      if (!this.expandSearch) {
        this.$tracker.event('Click', 'Live Listings Page: Clear Filters', 'Find a Project Page')
        // clear settings
        each(this.filters, filter => {
          this.settings[filter.type] = []
        })
 
        // clear counts
        this.filters = map(this.filters, filter => assign({}, filter, { count: 0 }))
 
        // clear search term
        this.searchTerm = ''
        this.updateSearchTerm(this.searchTerm)
 
        // update states
        this.updateSettings(this.settings)
        // click event
        if (typeof ignoreFetch === 'object') {
          this.$emit('fetch-projects', { init: 1, page: 1 })
          return
        }
 
        !ignoreFetch && this.$emit('fetch-projects')
      }
    },
    ...mapActions([
      'fetchSettings',
      'updateSettings',
      'storeSaveSettings',
      'resetIsSettingsSaved',
      'updateSearchTerm'
    ])
  }
}
</script>
