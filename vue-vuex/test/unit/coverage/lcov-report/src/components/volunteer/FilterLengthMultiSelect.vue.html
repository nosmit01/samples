<template>
  <!-- Lots of pixel pushing here to match the designs but not break the FilterMultiSelect UI
       that uses the same class names.
  -->
  <div
    class="caf-filter-bar-multiselect p-0 mt-0 mb-0"
  >
    <ul class="d-flex flex-column flex-md-row"
        :class="{'m-0': !isMobile}">
      <li
        v-for="(item, i) in items"
        :key="item.id"
        class="w-50"
        :class="{'border-right' : !isMobile && i === 0}"
        :style="{'margin-right' : !isMobile && i === 0 ? '30px' : 0}"
      >
        <label
          :for="item.id"
          class="vol-options length-filter d-flex flex-column justify-content-center px-2 px-md-0 mb-0 h-100"
        >
          <div class="d-flex align-items-center">
            <input type="checkbox" :id="item.id" v-model="selected" :value="item.id" @click="applySelection" />
            <p
              class="font-weight-bold label m-0 ml-2"
              :class="{'text-caf-red' : i === 0, 'text-caf-blue-accent' : i === 1}"
            >
              {{ item.name }}
            </p>
          </div>
          <div
            class="font-size-14 ml-4"
            :class="{'pr-1' : !isMobile && i === 0}"
          >
            {{ item.desc }}
          </div>
          <span
            v-if="i === 0"
            class="text-caf-red caf-filter-bar-multiselect-icoBG length-filter"
            style="top: -75px"
            :style="{'right' : isMobile ? '-20px' : 0}"
          >
            <ico-call height="100%" width="133px" />
          </span>
          <span
            v-if="i === 1"
            class="text-caf-blue-accent caf-filter-bar-multiselect-icoBG length-filter"
            style="bottom: -50px"
            :style="{'right' : isMobile ? '22px' : 0}"
          >
            <ico-briefcase height="100%" width="173px" />
          </span>
        </label>
      </li>
    </ul>
  </div>
</template>
 
<script>
import icoBriefcase from '@/assets/IcoBriefcase'
import icoCall from '@/assets/IcoCall'
 
export default {
  props: {
    items: {
      type: Array,
      required: true
    },
    settings: {
      type: Array,
      required: true
    },
    isMobile: {
      type: Boolean,
      default () {
        return false
      }
    }
  },
  data () {
    return {
      selected: []
    }
  },
  mounted () {
    this.setSelected()
  },
  components: {
    icoBriefcase,
    icoCall
  },
  methods: {
    applySelection () {
      this.$nextTick(() => {
        this.$emit('apply-selection', 'length', this.selected)
      })
    },
    close () {
      this.selected = []
      this.$emit('menu-open', false)
    },
    setSelected () {
      if (!this.settings) {
        return
      }
 
      this.selected = this.settings
    }
  },
  watch: {
    type () {
      this.setSelected()
    },
    settings () {
      this.setSelected()
    }
  }
}
</script>
