<template>
  <div>
    <hero />
    <filter-bar
      ref="filterBar"
      @fetch-projects="startFetchProjects"
    />
 
    <!-- See startFetchProjects() below for explanation -->
    <div ref="beforeGrid" class="mt-2 mt-md-3 pt-md-1"></div>
 
    <div class="container pt-1 pt-md-0 mb-3 pb-1 pb-md-0 text-center text-md-left">
      <div class="row">
        <div class="col-md-7 d-flex align-items-center justify-content-center justify-content-md-between pb-1 mb-2 pb-md-0 mb-md-0 order-md-0 order-1 hidden-print">
          <h6 v-if="loading" class="h5-md">
            Loading projects...
          </h6>
          <h6 v-if="!loading && pageObj && pageObj.count > 0" class="h5-lg pl-md-1 pl-lg-0 mb-0">
            Showing
            <span class="font-weight-bold">{{ startNumber }}-{{ startNumber + (projects.length - 1) }}</span>
            of
            <span class="font-weight-bold">{{ pageObj.count }}</span> opportunities
            <span v-if="searchTerm"> for &quot;{{ searchTerm }}&quot;</span>
          </h6>
        </div>
        <div
          class="col-md-5 d-flex align-items-center justify-content-md-end justify-content-center order-md-0 order-3 hidden-print"
          v-if="!searchTerm"
        >
          <div class="d-flex align-items-center justify-content-md-end justify-content-center w-100 pr-md-1 pr-lg-0" v-if="sortOptions.length > 1 && !$route.query.q">
            <span class="text-right text-nowrap">Sort by</span>
            <select
              id="sortByDropdown"
              class="form-control ml-2 mr-0"
              v-model="selectedSortOption"
              @change="reorder"
            >
              <option :value="option.value" :key="option.value" v-for="option in sortOptions">{{ option.label }}</option>
            </select>
          </div>
        </div>
      </div>
    </div>
 
    <div v-if="noResults" class="container">
      <div class="py-md-4 py-0 my-6 mx-md-0 px-2">
        <div v-if="!searchTerm" class="text-center">
          <h5 class="font-weight-semibold">
            Sorry, we couldn't find opportunities based on your selections.
          </h5>
          <h6>
            Try selecting different cause or skill areas, or explore some of our recently posted opportunities below.
          </h6>
        </div>
        <div v-if="searchTerm" class="text-center">
          <h5 class="font-weight-semibold">
            Sorry, there were no results found for "{{ searchTerm }}"
          </h5>
          <h6>
            Try another term or search by
            <span
              class="text-caf-red"
              style="cursor: pointer"
              @click="$refs.filterBar.setCollapseSearch">
              causes & skills
            </span>.
            Or, explore some of our recently posted opportunities below.
          </h6>
        </div>
      </div>
    </div>
 
    <app-grid-list
      type="volunteer"
      :cards="projects"
      :loading="loading"
      :page="pageObj"
      @previous-page="previousPage"
      @next-page="nextPage"
      @goto-page="gotoPage"
      :usePagination="!noResults"
      :perPage="perPage"
    >
      <template slot="no-results-text">
        <span v-if="searchTerm">
          Sorry, there were no results found for <strong>"{{ searchTerm }}"</strong>.
          <br class="hidden-md-down" />Try another term or filter projects by Category.
        </span>
      </template>
 
      <template slot="grid">
        <div class="col-md-4" v-for="project in projects" :key="project.id">
          <project-card
            :project="project"
            :tracking="{label: 'Live Listings Card ', category: 'Find a Project Page'}"
            class="mb-3"
          ></project-card>
        </div>
      </template>
 
    </app-grid-list>
  </div>
</template>
 
<script>
import { createNamespacedHelpers } from 'vuex'
 
import Hero from '@/components/volunteer/Hero'
import FilterBar from '@/components/volunteer/FilterBar'
import AppGridList from '@/components/AppGridList'
import ProjectCard from '@/components/volunteer/ProjectCard'
 
const { mapGetters, mapActions } = createNamespacedHelpers('volunteer')
const { mapGetters: mapBreakpointGetters } = createNamespacedHelpers('breakpoints')
 
export default {
  components: {
    Hero,
    FilterBar,
    AppGridList,
    ProjectCard
  },
  metaInfo: {
    title: 'Volunteer'
  },
  mounted () {
    !!this.firstLoad && this.fetchProjects()
    this.$tracker.event('View', 'Live Listings Page', 'Find a Project Page')
  },
  filters: {
  },
  data () {
    return {
      perPage: 18,
      filterBarFixed: false,
      paginate: false,
      firstLoad: !this.$route.query.hasOwnProperty('page'),
      sortOptions: [
        {
          value: 'recommended',
          label: 'Recommended'
        },
        {
          value: 'recent',
          label: 'Most Recent'
        },
        {
          value: 'name',
          label: 'Name (A-Z)'
        }
      ],
      selectedSortOption: this.$route.query.order || 'recommended'
    }
  },
  computed: {
    ...mapGetters([
      'loading',
      'projects',
      'noResults',
      'pageObj',
      'searchTerm'
    ]),
    ...mapBreakpointGetters(['isDesktop', 'isMobile']),
    startNumber () {
      return ((this.perPage * this.pageObj.number) - this.perPage) + 1
    }
  },
  beforeRouteEnter (to, from, next) {
    next(vm => {
      vm.setQueryString()
      next()
    })
  },
  methods: {
    ...mapActions([
      'initialize',
      'fetchProjects'
    ]),
    setQueryString () {
      this.$router.replace({
        query: Object.assign({}, this.$route.query)
      })
    },
    handleRouteChange () {
      if (!this.$route.query.hasOwnProperty('order')) {
        // manually reset the selected sort option
        this.selectedSortOption = 'recommended'
      }
      if (this.paginate) {
        this.paginate = false
      } else {
        !!this.$route.query.hasOwnProperty('page') && this.startFetchProjects(this.$route.query)
      }
    },
    startFetchProjects (params) {
      /* Solution for the following spec:
        If filters or search is changed, page is refreshed to page 1 with the selection bar at the
        top and the top row of results below, as per design spec (meaning, the page refreshes to
        page 1, but scrolled down so that the filter bar sits at the top)
      */
      this.$nextTick(() => {
        const filterBar = this.$refs.filterBar.$el
        if (filterBar) {
          const filterBarRect = filterBar.getBoundingClientRect()
          if (filterBarRect.top <= 0) {
            const el = this.$refs.beforeGrid
            if (el) {
              let scrollEl = 'html'
              if (/Safari/g.test(window.navigator.userAgent)) { scrollEl = 'html, body' }
              this.$jq(scrollEl).animate({ // animate into view
                scrollTop: this.$jq(el).offset().top - 70
              }, 500, () => {
                this.fetchProjects(params) // fetch after animation completes
                this.$router.push({
                  name: 'volunteer:projects',
                  query: params
                })
              })
            } else {
              this.fetchProjects(params)
            }
          } else { // if filterBar is not fixed to top, then still fetch
            this.fetchProjects(params)
          }
        } else { // if filterBar can't be resolved, then still fetch
          this.fetchProjects(params)
        }
        this.$router.push({
          name: 'volunteer:projects',
          query: params
        })
      })
    },
    previousPage () {
      this.paginate = true
      this.startFetchProjects({
        ...this.$route.query,
        page: this.pageObj.number === 1 ? 1 : this.pageObj.number - 1
      })
    },
    nextPage () {
      this.$tracker.event('Click', 'Live Listings Page: Next Page', 'Find a Project Page')
      this.paginate = true
      this.startFetchProjects({
        ...this.$route.query,
        page: this.pageObj.number + 1
      })
    },
    gotoPage (pageNum) {
      this.$tracker.event('Click', 'Live Listings Page: Goto Page', 'Find a Project Page')
      this.paginate = true
      this.startFetchProjects({
        ...this.$route.query,
        page: pageNum,
        searchTerm: this.searchTerm
      })
    },
    reorder (order) {
      let order_label
      switch (this.selectedSortOption) {
        case 'name':
          order_label = 'A-Z'
          break
        case 'recent':
          order_label = 'Most Recent'
          break
        case 'recommended':
          order_label = 'Recommended'
          break
      }
      this.$tracker.event('Click', 'Live Listings Page: Sort - ' + order_label, 'Find a Project Page')
      this.paginate = true
      this.startFetchProjects({
        ...this.$route.query,
        order: this.selectedSortOption,
        page: 1
      })
    }
  },
  watch: {
    $route: {
      handler: 'handleRouteChange',
      immediate: true
    },
    searchTerm (nVal, oVal) {
      if (!nVal && oVal) {
        // reset our sort to 'Recommended' when search is reset
        this.selectedSortOption = 'recommended'
      }
    }
  }
}
</script>
