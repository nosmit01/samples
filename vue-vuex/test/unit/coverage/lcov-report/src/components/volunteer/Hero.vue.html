<template>
  <header
    class="bg-color-gray-darkest py-5 volunteer-hero hidden-print"
    :style="{ background: 'url(' + bgImg + ') no-repeat left center', backgroundSize: 'cover' }"
  >
    <div class="container d-flex flex-column align-items-center justify-content-center h-100 w-100 px-3">
      <h3 class="text-color-white text-center mb-1 font-weight-normal">
        Find your next volunteer opportunity
      </h3>
      <h6 class="text-color-white text-center font-weight-medium px-3">
        Whether you have 1 hour to give or 20, organizations need professionals with skills like yours.
      </h6>
    </div>
  </header>
</template>
 
<script>
import { createNamespacedHelpers } from 'vuex'
 
import heroImg from '__STATIC__/images/responsive/background/volunteer_hero3X.jpg'
import heroImgMobile from '__STATIC__/images/responsive/background/volunteer_hero1X.jpg'
 
const { mapGetters: mapBreakpointGetters } = createNamespacedHelpers('breakpoints')
 
export default {
  data () {
    return {
      bgImg: heroImg
    }
  },
  mounted () {
    this.bgImg = (this.isMobile && window.screen.availWidth < 768) ? heroImgMobile : heroImg
  },
  computed: {
    ...mapBreakpointGetters(['isDesktop', 'isMobile'])
  }
}
</script>
