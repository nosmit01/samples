<template>
  <div class="caf-filter-bar-multiselect mt-4 mb-md-4 mb-0 px-2">
    <div class="mb-3 font-weight-bold text-caf-red">
      <label for="selectAll" class="select-all-btn m-0">Select All</label>
      <input class="invisible" id="selectAll" type="checkbox" v-model="selectAll" />
    </div>
    <ul>
      <li
        v-for="item in items"
        class="caf-filter-bar-multiselect-vol-options"
        :key="item.id"
      >
        <input
          type="checkbox"
          :id="item.id"
          v-model="selected"
          :value="item.id"
          :disabled="item.disabled"
          @click="applySelection"
        />
        <label
          :for="item.id"
          class="font-weight-semibold label mb-2 mb-md-1"
          :class="{'disabled' : item.disabled}"
        >
          {{ item.name }}
        </label>
      </li>
    </ul>
  </div>
</template>
 
<script>
import each from 'lodash/each'
import filter from 'lodash/filter'
 
export default {
  props: {
    items: {
      type: Array,
      required: true
    },
    type: {
      type: String,
      required: true
    },
    settings: {
      type: Array,
      required: true
    }
  },
  data () {
    return {
      selected: []
    }
  },
  mounted () {
    this.setSelected()
  },
  computed: {
    selectAll: {
      get: function () {
        // have to filter out disabled
        return this.items
          ? this.selected.length === filter(this.items, item => !item.disabled).length : false
      },
      set: function (value) {
        this.selected = []
 
        if (value) {
          each(this.items, item => {
            if (!item.disabled) {
              this.selected.push(item.id)
            }
          })
        }
 
        this.applySelection()
      }
    }
  },
  methods: {
    applySelection () {
      const type = this.type === 'causes' ? 'Causes' : 'Skills'
      this.$tracker.event('Click', `Live Listings Page: Select ${type}`, 'Find a Project Page')
      this.$nextTick(() => {
        this.$emit('apply-selection', this.type, this.selected)
      })
    },
    close () {
      this.selected = []
      this.$emit('menu-open', false)
    },
    setSelected () {
      if (!this.settings) {
        return
      }
 
      this.selected = this.settings
    }
  },
  watch: {
    type () {
      this.setSelected()
    },
    settings () {
      this.setSelected()
    }
  }
}
</script>
