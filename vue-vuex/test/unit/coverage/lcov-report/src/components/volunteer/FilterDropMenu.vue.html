<template>
  <div
    v-if="(open && !isSearchOpen) || (isMobile && isOpen && isSearchOpen)"
    class="caf-filter-bar-menu bg-color-white"
    :style="{'height' : !isMobile ? 'auto' : '100%'}"
    v-click-outside="setClose"
    v-sticky-top="{elTop : 50, scrollTop : 274, class: 'menu-fixed-top', isMobile, mobileClass: 'menu-fixed-mobile-top', bodyOffset : 50, isOpen}"
  >
    <div
      class="container p-0"
      :class="{'px-md-1' : isLengthFilter, 'h-100' : !isLengthFilter}"
    >
      <div class="col-12 px-0">
        <span
          v-if="!isMobile"
          class="caf-filter-bar-menu-closeIcon text-caf-red"
          :style="{top : isLengthFilter ? '30px' : '1px'}"
          @click="setClose"
        >
          <font-awesome-icon :icon="faTimesCircle" size="lg" />
        </span>
        <slot></slot>
      </div>
      <span v-if="bg && !isLengthFilter" class="text-caf-blue caf-filter-bar-multiselect-icoBG"
        :class="{'heart': bg.iconName==='heart'}">
        <font-awesome-icon :icon="bg" size="10x" />
      </span>
      <div
        v-if="isMobile"
        class="caf-filter-bar-menu-closeBtn bg-color-white p-3 w-100"
        :class="{'shadow' : !isLengthFilter}"
      >
        <button class="btn btn-secondary w-100" @click="setCloseBtn">Close</button>
      </div>
    </div>
  </div>
</template>
 
<script>
import { clickOutside, stickyTop } from '@/directives'
import { faTimesCircle } from '@fortawesome/pro-solid-svg-icons/faTimesCircle'
import FontAwesomeIcon from '@fortawesome/vue-fontawesome'
import { stickyTopMx } from '@/mixins'
 
export default {
  props: {
    isOpen: {
      type: Boolean,
      default () {
        return false
      }
    },
    type: { // can be any string. Checks if menu item is selected to keep menu open on outside click
      type: String,
      required: true,
      default () {
        return ''
      }
    },
    isMobile: {
      type: Boolean,
      default () {
        return false
      }
    },
    isSearchOpen: {
      type: Boolean,
      default () {
        return false
      }
    },
    bg: {
      type: Object
    }
  },
  data () {
    return {
      faTimesCircle,
      open: this.isOpen
    }
  },
  components: {
    FontAwesomeIcon
  },
  watch: {
    isOpen (newVal) {
      this.$nextTick(() => {
        this.open = newVal
      })
 
      setTimeout(() => { // next tick doesn't work here, needs more time for the DOM to be ready
        this.stickyTopMx({
          el: this.$el,
          elTop: 50,
          scrollTop: 274,
          stickyClass: 'menu-fixed-top',
          isMobile: this.isMobile,
          isMobileClass: 'menu-fixed-mobile-top',
          bodyOffset: 50,
          isOpen: this.open
        })
      }, 100)
    },
    isSearchOpen (newVal) {
      this.$nextTick(() => {
        if (newVal) this.setCloseBtn()
      })
    },
    // check if another filter is clicked to override closing of click outside
    type () {
      this.open = true
    }
  },
  directives: {
    clickOutside,
    stickyTop
  },
  mixins: [stickyTopMx],
  computed: {
    isLengthFilter () {
      return this.type === 'length'
    }
  },
  methods: {
    setClose () {
      const type = this.isMobile ? 'mobile-filter' : this.type
      this.open = false
      this.$nextTick(() => {
        if (this.isMobile && this.isSearchOpen) {
          return
        }
        if (!this.open) {
          this.$emit('menu-open', type)
        }
      })
    },
    setCloseBtn () {
      this.open = false
      this.$emit('menu-open', null)
    }
  }
}
</script>
