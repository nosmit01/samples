<template>
  <div
    class="caf-card-project card-listing"
    :class="{'phonecall' : project.is_phone_consultation, 'project' : !project.is_phone_consultation}"
    @mouseleave="stopTimer"
    @mouseover="trackHover"
  >
    <component
      ref="link"
      :is="'a'"
      :href="this.project.url"
      @click="trackClick"
      class="caf-card-project-container"
    >
      <div
        class="caf-card-project-header mb-0"
        :style="backgroundStyle"
      ></div>
 
      <div class="caf-card-project-body p-3 pt-4">
        <div v-if="get(project, 'sponsor.badge')" class="sponsor-tag">
          <img
            :src="get(project, 'sponsor.badge')"
            :alt="sponsoredBy"
          >
        </div>
 
        <div class="category-tag">
          <span v-if="project.is_phone_consultation">
            Phone call<span class="px-1">|</span>1 Hour
          </span>
          <span v-else>
            Project<span class="px-1">|</span>{{ project.timeframe }}
          </span>
        </div>
 
        <div class="caf-card-project-body-listingcontainer">
          <div class="caf-card-project-body-title d-flex align-items-center text-left">
            <h4 class="font-weight-semibold my-0">{{ project.name }}</h4>
          </div>
          <div class="caf-card-project-body-org pt-1 text-left text-color-gray">
            <em class="font-weight-normal text-black">for</em>
            {{ get(project, 'nonprofit.name') | truncate(52) }}
          </div>
        </div>
 
        <div class="caf-card-project-body-connection d-flex justify-content-between align-items-center text-left">
          <span>Share your skills with {{ get(project, 'volunteer_manager.name') }}.</span>
          <img
            v-if="get(project, 'volunteer_manager.image')"
            class="rounded-circle"
            :src="get(project, 'volunteer_manager.image')"
            :alt="get(project, 'volunteer_manager.name')"
            width="30"
            height="30"
          >
          <div
            v-else
            class="caf-card-project-body-connection-initials"
          >
            {{ get(project, 'volunteer_manager.initials') }}
          </div>
        </div>
 
        <div class="caf-card-project-body-impactstatement d-flex align-items-center text-color-gray text-left">
          <img :src="openQuoteSvg" alt="Impact Statement">
          <p v-if="project.is_phone_consultation" class="p-0 m-0">
            {{ project.topic | truncate(162) }}
          </p>
          <p v-else class="p-0 m-0">
            This project will save us {{ project.savings|currency }}, allowing us to {{ project.impact_statement | truncate(140) }}
          </p>
        </div>
 
        <div class="caf-card-project-body-cause d-flex align-items-center text-color-gray text-uppercase">
          {{ get(project, 'nonprofit.primary_cause') | truncate(36) }}
        </div>
      </div>
    </component>
  </div>
</template>
 
<script>
import FontAwesomeIcon from '@fortawesome/vue-fontawesome'
import faCalendarAlt from '@fortawesome/fontawesome-free-regular/faCalendarAlt'
import faClock from '@fortawesome/fontawesome-free-regular/faClock'
import callImg from '__STATIC__/images/responsive/background/back_card_phonecall.png'
import openQuoteSvg from '__STATIC__/images/responsive/icons/ico_quote_open.svg'
import get from 'lodash/get'
 
import { formatInteger, truncate, currency } from '@/filters'
 
export default {
  props: {
    project: {
      type: Object,
      required: true
    },
    tracking: {
      type: Object,
      required: false
    }
  },
  components: {
    FontAwesomeIcon
  },
  filters: {
    formatInteger,
    truncate,
    currency
  },
  mounted () {
  },
  data () {
    return {
      faCalendarAlt,
      faClock,
      callImg,
      openQuoteSvg,
      get
    }
  },
  computed: {
    backgroundStyle () {
      let style = {'background-image': '#324a9b'}
 
      if (get(this.project, 'show_cover_image')) {
        style = {'background': `url(${get(this.project, 'nonprofit.cover_image')})`}
      } else {
        style = {'background': `url(${callImg})`}
      }
 
      return style
    },
    sponsoredBy () {
      return `Sponsored by ${get(this.project, 'sponsor.name')}`
    }
  },
  methods: {
    trackHover () {
      if (this.tracking && !this.timer && !this.trackHoverComplete) {
        this.timer = window.setTimeout(() => {
          // double check in case of race condition
          if (!this.trackHoverComplete) {
            this.$tracker.event('Hover', this.tracking.label, this.tracking.category)
          }
          this.trackHoverComplete = true
          this.stopTimer()
        }, 1500)
      }
    },
    stopTimer () {
      if (this.tracking) {
        this.timer && window.clearTimeout(this.timer)
        this.timer = null
      }
    },
    trackClick () {
      if (this.tracking) {
        this.$tracker.event('Click', this.tracking.label, this.tracking.category)
      }
    }
  }
}
</script>
