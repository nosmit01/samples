<template>
  <div ref="gridlistContainer" class="caf-list-container">
    <div
      v-if="($route.query.view === 'list' && type !== 'bundles' && cards.length > 0) || showListView"
      class="container p-0 p-lg-1 caf-list"
    >
      <div class="caf-list-header">
        <div class="container d-flex caf-list-header-row">
          <slot name="list-header"></slot>
        </div>
      </div>
      <slot name="list"></slot>
    </div>
 
    <div class="container p-0 p-md-1 hidden-print" :class="{'loading-animation' : cards.length === 0 && loading}">
      <div v-if="$route.query.view !== 'list' || type === 'bundles'" class="row grid-mobile-margins" v-show="cards.length > 0">
        <slot name="grid"></slot>
      </div>
      <div
        class="row grid-mobile-margins"
        :class="{'mt-6' : !page}"
        v-if="cards.length === 0 && loading"
      >
        <div class="col-md-4" v-for="n in isMobile ? 3 : 9" :key="n">
          <app-loading-card></app-loading-card>
        </div>
      </div>
      <div class="row" v-if="cards.length === 0 && !loading">
        <div class="col-12">
          <div class="grid-listing-no-results p-3 mx-1 mx-md-0">
            <div class="text-center">
              <slot name="no-results-text">
                Sorry, we are unable to show any projects at the moment.
              </slot>
            </div>
          </div>
        </div>
      </div>
 
      <div class="caf-show-more mx-1 mx-md-0 mt-3 mt-md-0" v-show="page && page.has_next" @click="$emit('next-page')">
        <span v-show="!loading">Show More <font-awesome-icon :icon="faAngleDown" /></span>
        <span v-show="loading"><font-awesome-icon :icon="faCircleNotch" spin /></span>
      </div>
 
      <div
        v-if="!$route.query.view && type !== 'bundles'"
        class="caf-list-prefer-list"
        @click="setListView"
      >
        Prefer a list <span class="ico"></span> view?
      </div>
 
      <slot name="after-grid"></slot>
 
    </div>
  </div>
</template>
 
<script>
import { createNamespacedHelpers } from 'vuex'
import AppLoadingCard from '@/components/AppLoadingCard'
import FontAwesomeIcon from '@fortawesome/vue-fontawesome'
import faAngleDown from '@fortawesome/fontawesome-free-solid/faAngleDown'
import faCircleNotch from '@fortawesome/fontawesome-free-solid/faCircleNotch'
 
const { mapGetters: mapBreakpointGetters } = createNamespacedHelpers('breakpoints')
const print = window.matchMedia('print')
 
export default {
  props: {
    view: {
      type: Boolean,
      default () {
        return false
      }
    },
    cards: {
      type: Array,
      required: true
    },
    page: {
      type: Object,
      required: false
    },
    loading: {
      type: Boolean,
      default () {
        return false
      }
    },
    type: {
      type: String,
      required: false
    }
  },
  mounted () {
    // show list view when printing
    print.addListener(printing => {
      if (printing.matches) {
        this.showListView = true
      } else {
        setTimeout(() => {
          this.showListView = false
        }, 1000)
      }
    })
  },
  data () {
    return {
      faAngleDown,
      faCircleNotch,
      showListView: false
    }
  },
  components: {
    AppLoadingCard,
    FontAwesomeIcon
  },
  computed: {
    ...mapBreakpointGetters(['isDesktop', 'isMobile'])
  },
  methods: {
    setListView () {
      this.$tracker.event('Click', 'SF Menu List Callout', 'Storefront Menu')
      this.$emit('list-view')
      const offset = this.isMobile ? 120 : 80
      window.scrollTo(0, this.$refs.gridlistContainer.offsetTop - offset)
    }
  }
}
</script>
