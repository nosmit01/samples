import Vue from 'vue'
 
const store = {
  isMobile: false
}
 
const getters = {
  isMobile (state) {
    return state.isMobile
  },
  isDesktop (state) {
    return !state.isMobile
  }
}
 
const mutations = {
  setIsMobile (state, { isMobile }) {
    Vue.set(state, 'isMobile', isMobile)
  }
}
 
const actions = {
  setIsMobile ({ commit }, isMobile) {
    commit('setIsMobile', isMobile)
  }
}
 
export default {
  store,
  mutations,
  actions,
  getters,
  namespaced: true
}
